牛客的第六场。打得比较舒服的一场比赛。

T1 袜子分配

概率期望的结论题。考试时结论推错直接爆零

考虑每抽到一只袜子，再抽第二只颜色相同的概率为12n−1\frac{1}{2n-1}2n−11​，这也是抽到第二只期望的颜色相同次数。共取nnn次，那么根据期望的线性可加性，答案即为n2n−1\frac{n}{2n-1}2n−1n​。

int main(void) {
	int n;scanf("%d\n",&n);
	printf("%.10lf\n",double(n)/(2.*double(n)-1.));
	return 0;
}

T2 艰难睡眠

这题比赛上磕了一个半小时，最后才发现理解错了题意。考试考到一半才改题面的屑出题人

题意可以转化为在一个环上，选择某个长度特定的区间不能放置线段，放置nnn条线段使得代价最小。

考虑拆环为链并倍长，那么能放置的区间即为两不能放置的区间之间的区间。这个区间长度是固定的，那么用单调队列维护最小值即可。

复杂度O(nm)\mathcal{O}\left(nm\right)O(nm)。

const int maxn(5000+10),maxm(2000+10),inf(0x3f3f3f3f);
int n,m,k;
int a[maxn],b[maxn],c[maxn][maxm];
int q[maxn][maxm],head[maxn],tail[maxn];

inline int pos(const int &i) {
	return i > m ? i - m : i ;
}

int main(void) {
	cin>>n>>m>>k;
	for(int i(1);i<=n;i++) cin>>a[i]>>b[i];
	for(int i(1);i<=n;i++)
		for(int j(1);j<=m;j++)
			cin>>c[i][j];
	int ans(0);
	for(int i(1);i<=n;i++) {
		tail[i]=-1;
		for(int j(k+1);j+b[i]-1<=m;j++) {
			while(head[i]<=tail[i]&&c[i][q[i][tail[i]]]>c[i][j])
				tail[i]--;
			q[i][++tail[i]]=j;
		}
		if(head[i]<=tail[i])
			ans+=c[i][q[i][head[i]]];
		else {ans=inf;break;}
	}
	for(int l(2);l<=m;l++) {
		const int r(l+k-1);
		int res(0);
		for(int i(1);i<=n;i++) {
			if(l+m-b[i]>r) {
				while(head[i]<=tail[i]&&c[i][pos(q[i][tail[i]])]>c[i][pos(l+m-b[i])])
					tail[i]--;
				q[i][++tail[i]]=l+m-b[i];
			}
			while(head[i]<=tail[i]&&q[i][head[i]]<=r) head[i]++;
			if(head[i]<=tail[i])
				res+=c[i][pos(q[i][head[i]])];
			else {res=inf;break;}
		}
		ans=min(ans,res);
	}
	if(ans==inf) ans=-1;
	cout<<ans<<endl;
	return 0;
}

T3 路径难题

一见到图论一个激灵脑子就开窍了

先处理公交车。为每个公交车单独建点，站点向公交车连边，边权为车票费用，公交车向站点连边，边权为000。

然后考虑出租车。对于两点之间，假设搭乘出租车，那么在取整之前的最短路径在取整之后还是最短路径。（因为x<y↔⌈x⌉≤⌈y⌉x < y \leftrightarrow \lceil x \rceil \le \lceil y \rceilx<y↔⌈x⌉≤⌈y⌉）。

于是直接采用取整之前的费用跑最短路，在任意一条最短路径上重新取整算出费用。

为了使边权都为整数，出租车边的边权先设为ddd，公交车的为c×rc \times rc×r，跑完最短路后同一除以rrr即可。

const int maxn(2e5+10),maxk(1e4+10),maxt(4e5+10);
int n,m,k,q;
using ll = long long;
ll r;

struct Edge {
	int v,next;ll w;
	Edge(const int &v,const int &next,const ll &w)
		:v(v),next(next),w(w) {}
	Edge() {}
}e[(maxn<<1)+(maxt<<1)];
int head[maxn+maxk],cnt;
inline void AddEdge(const int &u,const int &v,const ll &w) {
	e[++cnt]=Edge(v,head[u],w);
	head[u]=cnt;
}

ll d[maxn+maxk];
int fa[maxn+maxk],fa_edge[maxn+maxk];
inline void Dijkstra(const int &s) {
	using pli=std::pair<ll,int>;
	static std::priority_queue<pli,std::vector<pli>,std::greater<pli>> q;
	static char vis[maxn+maxk];
	memset(d+1,0x3f,sizeof(ll)*(n+k));
	memset(vis+1,0x00,sizeof(char)*(n+k));
	fa[s]=fa_edge[s]=0;
	d[s]=0LL;q.emplace(d[s],s);
	while(!q.empty()) {
		int u(q.top().second);
		q.pop();
		if(!~vis[u]) continue;
		vis[u]=0xff;
		for(int i(head[u]);i;i=e[i].next) {
			const int &v(e[i].v);
			if(d[u]+e[i].w<d[v]) {
				d[v]=d[u]+e[i].w;
				fa[v]=u;fa_edge[v]=i;
				q.emplace(d[v],v);
			}
		}
	}
}

inline ll Solve(const int &s,const int &t) {
	Dijkstra(s);
	ll ans(0),now(0);
	for(int u(t);u;u=fa[u]) {
		if(u==s||fa[u]>n) {
			ans+=e[fa_edge[u]].w/r+ceil(double(now)/double(r));
			now=0;
		}
		else now+=e[fa_edge[u]].w;
	}
	return ans;
}

int main(void) {
	cin>>n>>m>>k>>r>>q;
	for(int i(1);i<=m;i++) {
		int u,v;ll w;cin>>u>>v>>w;
		AddEdge(u,v,w);
		AddEdge(v,u,w);
	}
	for(int i(1);i<=k;i++) {
		int t;ll c;
		cin>>t>>c;
		for(int j(1);j<=t;j++) {
			int x;cin>>x;
			AddEdge(x,n+i,0LL);
			AddEdge(n+i,x,c*r);
		}
	}
	for(int i(1);i<=q;i++) {
		int s,t;cin>>s>>t;
		cout<<Solve(s,t)<<endl;
	}
	return 0;
}

T4 牛半仙的妹子序列

考虑DP。设fif_ifi​表示经过iii的完美序列个数，那么有fi = ! size(nexti)+∑j∈nextifjf_i \ = \ ! \ size(next_i) + \sum_{j \in next_i} f_jfi​ = ! size(nexti​)+∑j∈nexti​​fj​，nextinext_inexti​表示在完美序列中能作为iii的后继的点集。现在的任务是找出所有的后继。

考虑jjj能作为iii的后继，当且仅当不存在kkk，使得i<k<j && pi<pk<pji < k < j \ \And\And \ p_i < p_k < p_ji<k<j && pi​<pk​<pj​，即所有在i,ji,ji,j之间的k , i<j<kk \ , \ i < j < kk , i<j<k都要满足pk<pi ∣∣ pk>pjp_k < p_i \ || \ p_k > p_jpk​<pi​ ∣∣ pk​>pj​。

为了找出所有的后继，首先找出在iii后面第一个值大于iii的位置，这个位置就是iii的第一个后继。

然后逐个找出下一个后继。考虑当前找出了iii的后继jjj，那么下一个后继kkk就要满足pi<pk<pjp_i < p _k < p_jpi​<pk​<pj​。满足这个条件且在最前面的位置即为下一个后继。

考虑使用权值线段树维护每个权值的最小位置。因为原序列是一个排列，所以线段树底层的值是唯一的。从右到左DP。

最后统计答案，如果一个位置没有作为任何一个位置的后继，就把它的fff值计入答案中。求和即可。

分析复杂度。考虑只有形如这样的序列才能让它寻找后继的次数达到最大值：1 6 5 4 3 2。设序列长度为nnn，那么它会在线段树上查询2n−12 n -12n−1次。即查询次数为O(n)\mathcal{O}(n)O(n)。其实是我找不出让它查询次数更多的序列了

const int maxn(2e5+10),inf(0x3f3f3f3f);
int n,a[maxn];
using ll = long long;
const ll mod(998244353);

namespace SegmentTree {
#define lc (a<<1)
#define rc ((a<<1)|1)
	int val[maxn<<2];
	void Build(const int &a=1,const int &l=1,const int &r=n) {
		val[a]=inf;
		if(l==r) return;
		int mid((l+r)>>1);
		Build(lc,l,mid);
		Build(rc,mid+1,r);
	}
	void Modify(const int &p,const int &k,const int &a=1,const int &l=1,const int &r=n) {
		if(l==r) return (void)(val[a]=k);
		int mid((l+r)>>1);
		if(p<=mid) Modify(p,k,lc,l,mid);
		else Modify(p,k,rc,mid+1,r);
		val[a]=min(val[lc],val[rc]);
	}
	int Query(const int &left,const int &right,const int &a=1,const int &l=1,const int &r=n) {
		if(right<l||left>r) return inf;
		if(left<=l&&right>=r) return val[a];
		int mid((l+r)>>1);
		return min(Query(left,right,lc,l,mid),Query(left,right,rc,mid+1,r));
	}
#undef lc
#undef rc
}

ll f[maxn];char vis[maxn];
int main(void) {
	cin>>n;
	SegmentTree::Build();
	for(int i(1);i<=n;i++) cin>>a[i];
	for(int i(n);i;i--) {
		int r(SegmentTree::Query(a[i]+1,n));
		if(r==inf) f[i]=1LL;
		for(;r!=inf;r=SegmentTree::Query(a[i]+1,a[r]-1)) {
			(f[i]+=f[r])%=mod;vis[r]=0xff;
		}
		SegmentTree::Modify(a[i],i);
	}
	ll ans(0);
	for(int i(1);i<=n;i++)
		if(~vis[i])
			(ans+=f[i])%=mod;
	cout<<ans<<endl;
	return 0;
}
