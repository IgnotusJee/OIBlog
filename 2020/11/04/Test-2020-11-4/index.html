2020-10-19考试总结

考试总结。

T1 下棋

大毒瘤记搜。开始以为是区间DP，写写调调一个半小时也没做出来。

其实是我把状态设错了。设f[i][j][k][p]表示放置i个白棋，j个黑棋，白棋最多比黑棋多k个，黑棋最多比白棋多p个的方案数。

const int maxn(155),maxm(25),mod(1e9+7);
int n,m,k;
int f[maxn][maxn][maxm][maxm];
int Dfs(const int &x,const int &y,const int &a,const int &b) {
    if(x==n&&y==m) return 1;
    if(~f[x][y][a][b]) return f[x][y][a][b];
    int ans(0);
    if(x<n&&a<k) (ans+=Dfs(x+1,y,a+1,max(0,b-1))%mod)%=mod;
    if(y<m&&b<k) (ans+=Dfs(x,y+1,max(0,a-1),b+1)%mod)%=mod;
    return f[x][y][a][b]=ans;
}

int main(void) {
    cin>>n>>m>>k;
    memset(f,0xff,sizeof f);
    cout<<Dfs(0,0,0,0)<<endl;
    return 0;
}

T2 出租车

也是一个DP为什么考这么多DP啊。考试时想的是把车上的人的目的地全压下来，然后DP，于是又TLE\texttt{TLE}TLE又WA\texttt{WA}WA。其实是因为这样无法处理多个人目的地相同的情况。

其实给三个人的位置分别开一维即可，剩下一位置直接上车就下一个人。

const int maxn(2000+10),maxk(11),inf(0x3f3f3f3f);
int n,k,u[maxn],v[maxn];

int f[maxn][maxk][maxk][maxk][maxk];
// 表示去接第i个人，在位置j，在车上的分别三个人要去哪里

int F(const int &i,const int &j,const int &a,const int &b,const int &c) {
    if(~f[i][j][a][b][c]) return f[i][j][a][b][c];
    int ans(inf);
    if(a) ans=min(ans,F(i,a,0,b,c)+abs(j-a)+1);
    if(b) ans=min(ans,F(i,b,a,0,c)+abs(j-b)+1);
    if(c) ans=min(ans,F(i,c,a,b,0)+abs(j-c)+1);// 先送一个人
    if(i>n) {
        if(!a&&!b&&!c) return 0;
        return f[i][j][a][b][c]=ans;
    }
    if(a&&b&&c) { //  出租车满了，先接一个人再让一个人滚回去
        ans=min(ans,F(i+1,a,v[i],b,c)+abs(j-u[i])+1+abs(u[i]-a)+1);
        ans=min(ans,F(i+1,b,a,v[i],c)+abs(j-u[i])+1+abs(u[i]-b)+1);
        ans=min(ans,F(i+1,c,a,b,v[i])+abs(j-u[i])+1+abs(u[i]-c)+1);
        ans=min(ans,F(i+1,v[i],a,b,c)+abs(j-u[i])+1+abs(u[i]-v[i])+1);
    }
    else { // 没满，再接一个人
        if(!a) ans=min(ans,F(i+1,u[i],v[i],b,c)+abs(j-u[i])+1);
        else if(!b) ans=min(ans,F(i+1,u[i],a,v[i],c)+abs(j-u[i])+1);
        else if(!c) ans=min(ans,F(i+1,u[i],a,b,v[i])+abs(j-u[i])+1);
    }
    return f[i][j][a][b][c]=ans;
}

int main(void) {
    cin>>n>>k;
    for(int i(1);i<=n;i++) cin>>u[i]>>v[i];
    memset(f,0xff,sizeof f);
    cout<<F(1,1,0,0,0)<<endl;
    return 0;
}

T3 堆石子

原题。考试时没想到优化，直接上暴搜了。其实是以前讲过的折半搜索。

int n,m,s[40],a[(int)(3e5+10)],b[(int)(3e5+10)],l,r,cnta,cntb;

int main(void) {
    cin>>n>>m;
    for(int i(1);i<=n;i++) cin>>s[i];
    l=n/2;r=n-l;
    for(int t(0);t<(1<<l);t++) {
        int now(0);
        for(int i(1);i<=l;i++)
            if(t&(1<<(i-1))) (now+=s[i]%m)%=m;
        a[++cnta]=now;
    }
    std::sort(a+1,a+1+cnta);
    cnta=std::unique(a+1,a+1+cnta)-a-1;
    for(int t(0);t<(1<<r);t++) {
        int now(0);
        for(int i(1);i<=r;i++)
            if(t&(1<<(i-1))) (now+=s[i+l]%m)%=m;
        b[++cntb]=now;
    }
    std::sort(b+1,b+1+cntb);
    cntb=std::unique(b+1,b+1+cntb)-b-1;
    int ans(0);
    for(int i(1),p;i<=cnta;i++) {
        p=std::upper_bound(b+1,b+1+cntb,m-a[i])-b-1;
        if(a[i]+b[p]<m&&p&&p<=cntb) ans=max(ans,a[i]+b[p]);
        p=std::upper_bound(b+1,b+1+cntb,2*m-a[i])-b-1;
        if(a[i]+b[p]<2*m&&p&&p<=cntb) ans=max(ans,a[i]+b[p]-m);
    }
    cout<<ans<<endl;
    return 0;
}

T4 不稳定的导弹系统

考试时没想到什么做法，觉得可能是网络流，最后因为时间不够就只能上暴搜了。

将每一个点裂成两个，被横向导弹打的和被纵向的导弹打的。横向的导弹向它能打的方向的下一个点，以此连下去，保证导弹的路径上的边权之和就是导弹终点的格点权值之和。纵向的相反。边权总和减去最小割即可。

const int maxn(55),inf(0x3f3f3f3f);
int n,m,map[maxn][maxn];

namespace Dinic {
    int s,t,N;

    struct Edge {
        int v,next,cap,flow;
        Edge(const int &v,const int &next,const int &cap,const int &flow=0)
            :v(v),next(next),cap(cap),flow(flow) {}
        Edge() {}
    }e[maxn*maxn*5<<1];
    int head[maxn*maxn*5],cnt=1;
    inline void AddEdge(const int &u,const int &v,const int &cap) {
        e[++cnt]=Edge(v,head[u],cap);
        head[u]=cnt;
        e[++cnt]=Edge(u,head[v],0);
        head[v]=cnt;
    }

    int d[maxn*maxn*5],cur[maxn*maxn*5];
    inline bool Bfs() {
        static std::queue<int> q;
        static char vis[maxn*maxn*5];
        memset(vis,0x00,sizeof(char)*N);
        d[s]=0;q.push(s);vis[s]=0xff;
        while(!q.empty()) {
            int u(q.front());q.pop();
            for(int i(head[u]);i;i=e[i].next) {
                const int &v(e[i].v);
                if(~vis[v]&&e[i].flow<e[i].cap) {
                    d[v]=d[u]+1;
                    q.push(v);
                    vis[v]=0xff;
                }
            }
        }
        return !~vis[t];
    }

    int Dfs(const int &u,int a) {
        if(!a||u==t) return a;
        int flow(0),f;
        for(int &i(cur[u]);i;i=e[i].next) {
            const int &v(e[i].v);
            if(d[u]+1==d[v]&&(f=Dfs(v,min(a,e[i].cap-e[i].flow)))>0) {
                e[i].flow+=f;e[i^1].flow-=f;
                flow+=f;a-=f;
            }
            if(!a) break;
        }
        return flow;
    }

    int Solve() {
        int flow(0);
        while(Bfs()) {
            memcpy(cur,head,sizeof(int)*N);
            flow+=Dfs(s,inf);
        }
        return flow;
    }
}

int tot;
int id[maxn][maxn][2];

int main(void) {
    cin>>n>>m;
    Dinic::s=++tot;
    Dinic::t=++tot;
    for(int i(1);i<=n;i++)
        for(int j(1);j<=m;j++) {
            cin>>map[i][j];
            id[i][j][0]=++tot;
            id[i][j][1]=++tot;
        }
    int ans(0);
    for(int i(1);i<=n;i++)
        for(int j(1);j<=m;j++) {
            Dinic::AddEdge(id[i][j][0],id[i][j][1],inf);
            int maxi(0);
            switch(map[i][j]) {
                case -1: {
                    Dinic::AddEdge(Dinic::s,id[i][j][0],inf);
                    for(int k(i);k;k--) maxi=max(maxi,map[k][j]);
                    for(int k(i);k>1;k--)
                        Dinic::AddEdge(id[k][j][0],id[k-1][j][0],maxi-max(map[k][j],0));
                    break;
                }
                case -2: {
                    Dinic::AddEdge(Dinic::s,id[i][j][0],inf);
                    for(int k(i);k<=n;k++) maxi=max(maxi,map[k][j]);
                    for(int k(i);k<n;k++)
                        Dinic::AddEdge(id[k][j][0],id[k+1][j][0],maxi-max(map[k][j],0));
                    break;
                }
                case -3: {
                    Dinic::AddEdge(id[i][j][1],Dinic::t,inf);
                    for(int k(j);k;k--) maxi=max(maxi,map[i][k]);
                    for(int k(j);k>1;k--)
                        Dinic::AddEdge(id[i][k-1][1],id[i][k][1],maxi-max(map[i][k],0));
                    break;
                }
                case -4: {
                    Dinic::AddEdge(id[i][j][1],Dinic::t,inf);
                    for(int k(j);k<=m;k++) maxi=max(maxi,map[i][k]);
                    for(int k(j);k<m;k++)
                        Dinic::AddEdge(id[i][k+1][1],id[i][k][1],maxi-max(map[i][k],0));
                    break;
                }
                default:;
            }
            ans+=maxi;
        }
    Dinic::N=tot+5;
    cout<<ans-Dinic::Solve()<<endl;
    return 0;
}
