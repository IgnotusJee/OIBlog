T1 劳孙肉饼

sb题。开始打表把暴力写错了，调暴力就花了半个多小时。打出表后找出了规律。问题还是时间分配不对，在第一题上太多。

另外，这题原来交上去爆long long了。以后还要考虑是否要龟速乘的问题。

typedef __int128_t ll;
const ll mod(1e9+7);
int main(void) {
    ll x,y;
    cin>>x>>y;
    if(x>y) swap(x,y);
    y--;
    if(x&1LL) cout<<(x%mod+(y%mod)*(((x+1LL)>>1)%mod)%mod)%mod<<endl;
    else cout<<(x%mod+(((y+1LL)>>1LL)%mod)*(((x+1LL)>>1LL)%mod)%mod+
        (((y>>1LL)%mod)*((((x+1)>>1LL)+1LL))%mod)%mod)%mod<<endl;
    return 0;
}

T2 未来程序

字符串模拟加拓扑排序。考试时花了大概一个小时就糊出来了。

其实考试时想复杂了。这种调用只可能有自环，并且函数给出的顺序满足拓扑序。可以直接求解。

const int maxn(1e4+10);
int T,n;
typedef unsigned long long ull;
const string VOI("void"),A("A"),END("}"),INTT("int"),PRI("printf");

map<string,int> mp;
int tot;

struct Edge {
    int v,next;
    Edge(const int &v,const int &next) :v(v),next(next) {}
    Edge() {}
}e[maxn<<5];
int head[maxn],k;
inline void AddEdge(const int &u,const int &v) {
    e[++k]=Edge(v,head[u]);
    head[u]=k;
}

ull cnt[maxn];
int ind[maxn];
char ha_cir;
ull Topo() {
    for(int u(1);u<=tot;u++)
        for(int i(head[u]);i;i=e[i].next) {
            const int &v(e[i].v);
            if(u==v) {
                if(cnt[u]) {ha_cir=0xff;return 0ULL;}
            }
            else ind[v]++;
        }
    static queue<int> q;
    for(int i(1);i<=tot;i++) if(!ind[i]) q.push(i);
    while(!q.empty()) {
        int u(q.front());q.pop();
        for(int i(head[u]);i;i=e[i].next) {
            const int &v(e[i].v);
            cnt[v]+=cnt[u];
            if(!--ind[v]) q.push(v);
        }
    }
    for(int i(1);i<=tot;i++) if(ind[i]&&cnt[i]) {ha_cir=0xff;break;}
    return cnt[1];
}

inline void Split(string &s) {
    for(int i(0);i<(int)s.length();i++) if(s[i]=='(') {
        s.erase(i);return;
    }
}

char s[50];
string now;
int main(void) {
    ios::sync_with_stdio(false);
    cin.tie();cout.tie();
    cin>>T;
    while(T--) {
        char flag(0x00);
        mp.clear();mp[A]=1;tot=1;
        memset(ind,0,sizeof ind);
        memset(cnt,0,sizeof cnt);
        memset(head,0,sizeof head);k=0;
        cin>>n;
        for(int i(1);i<=4;i++) cin.getline(s,50);
        while(cin>>now,now==VOI&&~flag) {
            cin>>now;
            Split(now);
            if(mp.find(now)!=mp.end()) {flag=0xff;break;}
            int u(mp[now]=++tot);
            while(cin>>now,now!=END&&~flag) {
                Split(now);
                if(mp.find(now)==mp.end()) {flag=0xff;break;}
                AddEdge(u,mp[now]);
            }
        }
        cin.getline(s,50);cin.getline(s,50);
        while(cin>>now) {
            Split(now);
            if(now==PRI) break;
            cnt[mp[now]]++;
        }
        if(!~flag) cout<<"Compile error\n";
        else {
            ha_cir=0x00;
            ull ans(Topo());
            if(!~ha_cir) cout<<"Segmentation fault\n";
            else cout<<ans<<endl;
        }
        cin.getline(s,50);cin.getline(s,50);
    }
    return 0;
}

另外，注意关闭流同步后不要调用<cstdio>中的任何函数。

T3 赛博朋克酒保行动

DAG上的期望DP。考试时花了一个多一点小时糊出来了，但因为建DAG时复杂度出问题导致100pts→\rightarrow→40pts。还是考CSP的教训：不要侥幸地觉得不确定的复杂度能过去。

const int maxn(1000+10),maxm(3000+10);
typedef long long ll;
ll mod(998244353);
int T,n,m,k,C;

struct Edge {
    int v,next;
    Edge(const int &v,const int &next) :v(v),next(next) {}
    Edge() {}
}e[maxm<<1];
int head[maxn],cnt;
inline void AddEdge(const int &u,const int &v) {
    e[++cnt]=Edge(v,head[u]);
    head[u]=cnt;
}
int c[maxn],a[maxn],b[maxn];
inline ll QuickPow(ll a,ll b) {
    ll ans(1);if(!b||a==1) return ans;
    for(;b;b>>=1,(a*=a)%=mod) if(b&1) (ans*=a)%=mod;
    return ans;
}

std::vector<int> G[maxn]; // 保存DAG
ll f[maxn][maxn],g[maxn][maxn];
// f[i][j] 表示第i支队伍走到j的概率，g[i][j]表示第i个势力经过j的概率
int d[maxn];
int ind[maxn],outd[maxn];

int vis[maxn];
bool Dfs(const int &u,const int &t) {
    if(u==t||vis[u]==1) return true;
    if(vis[u]==2) return false;
    bool flag(false);
    for(int i(head[u]);i;i=e[i].next) {
        const int &v(e[i].v);
        if(d[u]+1!=d[v]) continue;
        if(Dfs(v,t)) G[u].push_back(v),flag=true;
    }
    vis[u]=(flag?1:2);
    return flag;
}

inline void Bfs(const int &s,const int &t) {
    static std::queue<int> q;
    static char vis[maxn];
    memset(d,0x3f,sizeof d);
    memset(vis,0x00,sizeof vis);
    d[s]=0;q.push(s);vis[s]=0xff;
    while(!q.empty()) {
        int u(q.front());q.pop();
        for(int i(head[u]);i;i=e[i].next) {
            const int &v(e[i].v);
            if(~vis[v]) {
                d[v]=d[u]+1;
                q.push(v);
                vis[v]=0xff;
            }
        }
    }
    for(int i(1);i<=n;i++) G[i].clear();
    memset(ind,0,sizeof ind);
    memset(outd,0,sizeof outd);
    memset(::vis,0,sizeof ::vis);
    Dfs(s,t);
}

#define inv(x) QuickPow((x),mod - 2)
inline void Topo(const int &s,ll *f) {
    for(int u(1);u<=n;u++)
        for(const int &v: G[u])
            outd[u]++,ind[v]++;
    static std::queue<int> q;
    f[s]=1;q.push(s);
    while(!q.empty()) {
        int u(q.front());q.pop();
        for(const int &v: G[u]) {
            (f[v]+=f[u]*inv(outd[u])%mod)%=mod;
            if(!--ind[v]) q.push(v);
        }
    }
}

int main(void) {
    cin>>T;
    while(T--) {
        memset(head,0,sizeof head);cnt=C=0;
        memset(f,0,sizeof f);
        memset(g,0,sizeof g);
        cin>>n>>m>>k;
        for(int i(1);i<=m;i++) {
            int u,v;
            cin>>u>>v;
            AddEdge(u,v);AddEdge(v,u);
        }
        for(int i(1);i<=k;i++) {
            cin>>c[i]>>a[i]>>b[i];C=max(C,c[i]);
            Bfs(a[i],b[i]);
            Topo(a[i],f[i]);
            for(int j(1);j<=n;j++)
                g[c[i]][j]=((1-((1-g[c[i]][j])%mod+mod)%mod*((1-f[i][j])%mod+mod)%mod)%mod+mod)%mod;
        }
        for(int i(1);i<=n;i++) {
            ll res(0),sum(1),tmp(1);
            int cnt(0);
            for(int j(1);j<=C;j++) {
                if(g[j][i]!=1LL) (sum*=((1-g[j][i])%mod+mod)%mod)%=mod;
                else cnt++;
                (tmp*=((1-g[j][i])%mod+mod)%mod)%=mod;
            }
            if(cnt>1) sum=0LL;
            for(int j(1);j<=C;j++)
                if(g[j][i]!=1LL) (res+=g[j][i]*tmp%mod*inv(1-g[j][i])%mod)%=mod;
                else (res+=g[j][i]*sum%mod)%=mod;
            cout<<((1-(res+tmp)%mod)%mod+mod)%mod<<endl;
        }
    }
    return 0;
}

T4 月之美兔

考试时打T4时已经只剩半小时了。看了题觉得是原题就直接照着原题的方法去打了，结果连样例都过不去。不过还是拿了几分

考完后自己糊了个BFS+Dijkstra，居然跑的比标程还快。

const int maxn(2e4+10);
int n,m,s,t,ed;
struct Node {
    int i,j,step; // 当前频率为i , 调频为j
    bool flag;
    Node(const int &i,const int &j,const int &step,const bool &flag=false)
        :i(i),j(j),step(step),flag(flag) {}
    bool operator < (const Node &p) const {return this->step>p.step;}
};
std::queue<Node> q;
std::vector<int> local[maxn];
std::vector<Node> away[maxn];
std::bitset<maxn> vis[maxn];
std::unordered_map<int,std::unordered_map<int,int>> d;

inline void Insert(const int &x,const int &y,const int &z,const bool flag) {
    if(vis[x].test(y)) return;
    q.emplace(x,y,z,flag);
    vis[x].set(y);
    away[x].push_back(Node(x,y,z,flag));
}
void Bfs() {
    for(int i(0);i<n;i++)
        for(const int &j: local[i])
            Insert(i,j,0,i==t&&j==ed);
    while(!q.empty()) {
        Node u(q.front());q.pop();
        if(u.i - u.j >= 0) Insert(u.i - u.j, u.j, u.step + 1, u.flag);
        if(u.i + u.j < n) Insert(u.i + u.j, u.j, u.step + 1, u.flag);
    }
}

inline bool exist(const int &i,const int &j) {
    if(d.find(i)==d.end()) return false;
    if(d[i].find(j)==d[i].end()) return false;
    return true;
}
int Dijkstra() {
    static std::priority_queue<Node> q;
    for(int i(0);i<n;i++) vis[i].reset();
    for(const auto &u: away[s]) {
        q.push(u);
        d[u.i][u.j]=u.step;
    }
    while(!q.empty()) {
        Node u(q.top());q.pop();
        if(u.i == t || u.flag) return u.step;
        if(vis[u.i].test(u.j)) continue;
        vis[u.i].set(u.j);
        if(u.i - u.j >= 0) {
            for(const auto &e: away[u.i - u.j]) {
                Node v(u.i - u.j, e.j, u.step + e.step + 1, e.flag);
                if(!exist(v.i,v.j) || v.step < d[v.i][v.j]) {
                    d[v.i][v.j] = v.step;
                    q.push(v);
                }
            }
            Node v(u.i - u.j, u.j, u.step + 1, u.flag);
            if(!exist(v.i,v.j) || v.step < d[v.i][v.j]) {
                d[v.i][v.j] = v.step;
                q.push(v);
            }
        }
        if(u.i + u.j < n) {
            for(const auto &e: away[u.i + u.j]) {
                Node v(u.i + u.j, e.j, u.step + e.step + 1, e.flag);
                if(!exist(v.i,v.j) || v.step < d[v.i][v.j]) {
                    d[v.i][v.j] = v.step;
                    q.push(v);
                }
            }
            Node v(u.i + u.j, u.j, u.step + 1, u.flag);
            if(!exist(v.i,v.j) || v.step < d[v.i][v.j]) {
                d[v.i][v.j] = v.step;
                q.push(v);
            }
        }
    }
    return 114514;
}

int main(void) {
    cin>>n>>m;
    for(int i(1);i<=m;i++) {
        int a,f;cin>>a>>f;
        if(i==1) s=a;
        if(i==2) t=a,ed=f;
        local[a].push_back(f);
    }
    if(s==t) {cout<<"0\n";return 0;}
    Bfs();
    cout<<Dijkstra()<<endl;
    return 0;
}
