void Dijkstra(const int &s=::s) {
    typedef std::pair<ll,int> pii;
    static std::priority_queue< pii, std::vector<pii>, std::greater<pii> > q;
    static char vis[maxn];
    memset(d+1,0x3f,sizeof(ll)*n);
    memset(vis+1,0x00,sizeof(char)*n);
    d[s]=0LL;q.push(std::make_pair(d[s],s));
    while(!q.empty()) {
        int u=q.top().second;
        q.pop();
        if(!~vis[u]) continue;
        vis[u]=0xff;

            const int &v=e[i].v;
            if(d[u]+e[i].w<d[v]) {
                d[v]=d[u]+e[i].w;
                q.push(std::make_pair(d[v],v));
            }
        }
    }

class RMQLCA {
private:
    int fir[maxn],len;
    int p[maxn<<1][19],Log[maxn<<1];
    void Dfs(const int &u=1) {
        p[++len][0]=u;
        fir[u]=len;
        for(int i(head[u]);i;i=e[i].next) {
            const int &v=e[i].v;
            if(v==fa[u]) continue;
            Dfs(v);
            p[++len][0]=u;
        }
    }
    static inline int GetMin(const int &u,const int &v) {
        if(!u||!v) return u|v;
        return depth[u]<depth[v]?u:v;
    }
    void ST_Init() {
        for(int i(0),w(1);w<=len;i++,w<<=1) Log[w]=i;
        for(int i(1);i<=len;i++) if(!Log[i]) Log[i]=Log[i-1];
        for(int j(1);j<19;j++)
            for(int i(1);i<=len;i++)
                if(i+(1<<j)-1<=len)
                    p[i][j]=GetMin(p[i][j-1],p[i+(1<<(j-1))][j-1]);
    }
public:
    void Init() {
        Dfs();ST_Init();
    }
    inline int operator () (const int &u,const int &v) {
        if(!u||!v) return 0;
        int l(fir[u]),r(fir[v]);
        if(l>r) swap(l,r);
        int o(Log[r-l+1]);
        return GetMin(p[l][o],p[r-(1<<o)+1][o]);
    }

void Tarjan(const int &u,const int &fa=0) {
    dfn[u]=low[u]=++stamp;
    int cnt_ch=0;
    for(int i(head[u]);i;i=e[i].next) {
        const int &v=e[i].v;
        if(v==fa) continue;
        if(!dfn[v]) {
            stack[++top]=std::make_pair(u,v);
            cnt_ch++;
            Tarjan(v,u);
            low[u]=min(low[u],low[v]);
            if(low[v]>=dfn[u]) {
                iscut[u]=0xff;
                bcc[++cnt_bcc].clear();
                for(;;) {
                    const pii &s=stack[top--];
                    if(belong[s.first]!=cnt_bcc) {
                        bcc[cnt_bcc].push_back(s.first);
                        belong[s.first]=cnt_bcc;
                    }
                    if(belong[s.second]!=cnt_bcc) {
                        bcc[cnt_bcc].push_back(s.second);
                        belong[s.second]=cnt_bcc;
                    }
                    if(s.first==u&&s.second==v) break;
                }
            }
        }
        else if(dfn[u]>dfn[v]) {
            stack[++top]=std::make_pair(u,v);
            low[u]=min(low[u],dfn[v]);
        }
    }
    if(!fa&&cnt_ch==1) iscut[u]=0x0;
}

PrimPrimPrim

  • O(n2)O(n^2)O(n2)
const int maxn=2e3+20,inf=0x3f3f3f3f;
int n;
ll p[maxn][maxn];

ll prim(const int &s) {
    static int cloest[maxn];
    static ll dist[maxn];
    ll ans=0;dist[n+1]=inf;
    for(int i(0);i<=n;i++) {
        if(i==s) dist[i]=0,cloest[i]=-1;
        else dist[i]=p[s][i],cloest[i]=s;
    }
    for(int i(1);i<=n;i++) {
        int now=n+1;
        for(int j(0);j<=n;j++)
            if(~cloest[j]&&dist[j]<dist[now])
                now=j;
        ans+=dist[now];
        dist[now]=0;cloest[now]=-1;
        for(int j(0);j<=n;j++)
            if(~cloest[j])
                dist[j]=min(dist[j],p[now][j]);
    }
    return ans;
}

int map[maxn][maxn];
int n,m;
int w[maxn];
bool visx[maxn],visy[maxn];
int lx[maxn],ly[maxn],slack[maxn],match[maxn];
bool dfs(int u) {
    visx[u]=true;
    for(int v=1;v<=n;v++) {
        if(visy[v]) continue;
        int delta=lx[u]+ly[v]-map[u][v];
        if(!delta) {//可以增广
            visy[v]=true;
            if(!match[v] || dfs(match[v])) {
                match[v]=u;
                return true;
            }
        }
        else slack[v]=min(slack[v],delta);
    }
    return false;
}
void update() {
    int d=inf;
    for(int i=1;i<=n;i++)
        if(!visy[i])
            d=min(d,slack[i]);
    for(int i=1;i<=m;i++)
        if(visx[i]) lx[i]-=d;
    for(int i=1;i<=n;i++) {
        if(visy[i]) ly[i]+=d;
        else slack[i]-=d;
    }
}
int km() {
    clr(lx),clr(ly),clr(match);
    for(int u=1;u<=m;u++) {
        lx[u]=map[u][1];
        for(int v=2;v<=n;v++)
            lx[u]=max(lx[u],map[u][v]);
    }
    for(int i=1;i<=m;i++) {
        clrx(slack);
        while(true) {
            clr(visx),clr(visy);
            if(dfs(i)) break;
            update();
        }
    }
    int ans=0;
    for(int v=1;v<=n;v++)
        ans-=map[match[v]][v];
    return ans;
}
