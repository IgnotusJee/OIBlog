const int maxn=5e4+5,inf=0x3f3f3f3f;
int n,m,k;
int a[maxn],belong[maxn],size;
int now,cnt[maxn];

struct query {
    int l,r,id;
}q[maxn];
int ans[maxn];
inline bool cmp(const query &a,const query &b) {
    return belong[a.l]^belong[b.l] ? a.l<b.l : (belong[a.l]&1 ? a.r<b.r : a.r>b.r);
}
inline void add(int x) {
    cnt[a[x]]++;
    now+=2*cnt[a[x]]-1;
}
inline void del(int x) {
    cnt[a[x]]--;
    now-=2*cnt[a[x]]+1;
}

int main(void) {
    read(n,m,k);size=n/sqrt(m);
    for(int i(1);i<=n;i++) {
        read(a[i]);
        belong[i]=(i-1)/size+1;
    }
    for(int i(1);i<=m;i++) {
        read(q[i].l,q[i].r);
        q[i].id=i;
    }
    std::sort(q+1,q+m+1,cmp);
    int l=1,r=0;
    for(int i(1);i<=m;i++) {
        while(l<q[i].l) del(l++);
        while(l>q[i].l) add(--l);
        while(r<q[i].r) add(++r);
        while(r>q[i].r) del(r--);
        ans[q[i].id]=now;
    }
    for(int i(1);i<=m;i++) write(ans[i],'\n');
    flush();
    return 0;
}

回滚莫队

只增莫队:对于左端点同块的询问按右端点单增排序,莫队的右端点和左端点都移到当前块的右端点上,对于每一个询问的右端点就直接向后增加进集合并更新当前答案,询问的左端点就复制一份当前的答案,向后增加至询问的左端点后更新进答案数组,然后删除回滚,不贡献答案.

只减莫队:将左端点同块的询问按右端点单减排序,莫队的右端点先移到 nnn ,对于每一个询问的右端点直接减去并更新当前的答案,莫队的左端点先移到当前块的左端点,每次询问的左端点就复制一份当前的答案,减到询问的左端点后更新进答案数组,然后增加回滚,不贡献答案.

namespace CaptainMo {
#define lc(i) ((i-1)*size+1)
#define rc(i) min(n,i*size)
    int belong[maxn],size;
    struct Query {
        int l,r,id;
        Query(const int &l=0,const int &r=0,const int &id=0)
            :l(l),r(r),id(id) {}
    }q[maxn];
    struct Compare {
        bool operator () (const Query &a,const Query &b) {
            return (belong[a.l] ^ belong[b.l]) ? (belong[a.l] < belong[b.l]) : (a.r < b.r);
        }
    }cmp;

    int Lloc[maxn],Rloc[maxn],cnt[maxn];
    void build_block() {
        size=sqrt(n);
        for(int i(1);i<=n;i++) belong[i]=(i-1)/size+1;
        std::sort(q+1,q+1+m,cmp);
        clr(Lloc,0x3f),clr(Rloc,0xc1);
    }
    inline void ins(const int &x,int &ans) {
        Lloc[a[x]]=min(Lloc[a[x]],x);
        Rloc[a[x]]=max(Rloc[a[x]],x);
        cnt[a[x]]++;
        ans=max(ans,abs(Rloc[a[x]]-Lloc[a[x]]));
    }
    inline void del(const int &x) {
        Lloc[a[x]]=inf;
        Rloc[a[x]]=nef;
        if(cnt[a[x]]) cnt[a[x]]--;
    }
    inline void insl(const int &x,int &ans) {
        if(Rloc[a[x]]==nef) Rloc[a[x]]=x;
        cnt[a[x]]++;
        ans=max(ans,abs(Rloc[a[x]]-x));
    }
    inline void dell(const int &x) {
        if(!--cnt[a[x]]) Rloc[a[x]]=nef;
    }
    inline int calc(const int &l,const int &r) {
        static int last[maxn];//记录每一个数第一次出现的位置
        int ans=0;
        for(int i(l);i<=r;i++) {
            if(!last[a[i]]) last[a[i]]=i;
            else ans=max(ans,i-last[a[i]]);
        }
        for(int i(l);i<=r;i++) last[a[i]]=0;
        return ans;
    }
    int ans[maxn];
    void solve() {
        static int l=0,r=n+1,now_ans,last_block=0,last_ans;
        for(int i(1);i<=m;i++) {
            if(belong[q[i].l]==belong[q[i].r])
                ans[q[i].id]=calc(q[i].l,q[i].r);
            else {
                if(belong[q[i].l]!=last_block) {
                    while(r>rc(belong[q[i].l])) del(r--);
                    while(l<r+1) del(l++);
                    last_block=belong[q[i].l];
                    last_ans=nef;
                }
                while(r<q[i].r) ins(++r,last_ans);
                now_ans=last_ans;
                while(l>q[i].l) insl(--l,now_ans);
                ans[q[i].id]=now_ans;
                while(l<=rc(belong[q[i].l])) dell(l++);
            }
        }
    }
#undef lc
#undef rc
}

inline void modify(const ll l,const ll r,const ll add) {
    for(ll i=l;i<=Min(number[l]*len,r);i++)
        a[i]+=add,sum[number[i]]+=add;
    if(number[l]!=number[r])
        for(ll i=(number[r]-1)*len+1;i<=r;i++)
            a[i]+=add,sum[number[i]]+=add;;
    for(ll i=number[l]+1;i<number[r];i++)
        delta[i]+=add;
}
inline ll query(const ll l,const ll r,const ll x) {
    ll ans=0;
    for(ll i=l;i<=Min(number[l]*len,r);i++)
        ans=(ans+a[i]+delta[number[i]])%(x+1);
    if(number[l]!=number[r])
        for(ll i=(number[r]-1)*len+1;i<=r;i++)
            ans=(ans+a[i]+delta[number[i]])%(x+1);

        ans=(ans+sum[i]+len*delta[i])%(x+1);
    return ans%(x+1);
}

const int maxn=1.5e5+10,inf=0x3f3f3f3f;
int n,m;

namespace LeftLearningTree {
#define lc st[x].ch[0]
#define rc st[x].ch[1]
    struct Node {
        int ch[2],val,dist,root;
    }st[maxn];
    int Merge(int x,int y) {
        if(!x||!y) return x|y;
        if(st[x].val>st[y].val||(st[x].val==st[y].val&&x>y))
            swap(x,y);
        rc=Merge(rc,y);
        if(st[lc].dist<st[rc].dist)
            swap(lc,rc);
        st[lc].root=st[rc].root=st[x].root=x;
        st[x].dist=st[rc].dist+1;
        return x;
    }
    int GetRoot(const int &x) {
        return st[x].root == x ? x : st[x].root = GetRoot(st[x].root);
    }
    inline void Pop(const int &x) {
        st[x].val=-1;
        st[lc].root=lc;
        st[rc].root=rc;
        st[x].root=Merge(lc,rc);
    }
#undef lc
#undef rc
}
using namespace LeftLearningTree;

int main(void) {
    read(n,m);
    st[0].dist=-1;
    for(int i(1);i<=n;i++) {
        st[i].root=i;
        read(st[i].val);
    }
    for(int i(1);i<=m;i++) {
        int opt,x,y;
        read(opt);
        if(opt==1) {
            read(x,y);
            if(!~st[x].val||!~st[y].val) continue;
            int u=GetRoot(x);
            int v=GetRoot(y);
            if(u==v) continue;
            st[u].root=st[v].root=Merge(u,v);
        }
        else if(opt==2) {
            read(x);
            if(!~st[x].val) write("-1\n");
            else write(st[GetRoot(x)].val,'\n'),Pop(GetRoot(x));
        }
    }
    flush();
    return 0;
}

typedef long long ll;
const ll maxn=1e6+50;
ll n,m,c1[maxn],c2[maxn];
inline ll lowbit(ll x) {
    return x&(-x);
}
inline void modify(ll x,ll y,ll *c) {
    for(;x<=n;x+=lowbit(x)) c[x]+=y;
}
inline ll sum(ll x,ll *c) {
    ll ans=0;
    for(;x;x-=lowbit(x)) ans+=c[x];
    return ans;
}
inline void augment(ll x,ll y,ll z) {
    modify(x,z,c1);
    modify(y+1,-z,c1);
    modify(x,z*x,c2);
    modify(y+1,-z*(y+1),c2);
}
inline ll query(ll x,ll y) {
    return (y+1)*sum(y,c1)-sum(y,c2)-x*sum(x-1,c1)+sum(x-1,c2);
}
int main(void) {
    read(n),read(m);
    for(register int i=1;i<=n;i++) {
        register ll a;
        read(a);
        augment(i,i,a);
    }
    for(register int i=1;i<=m;i++) {
        register ll opt,l,r;
        read(opt),read(l),read(r);
        if(opt==1) {
            register ll x;
            read(x);
            augment(l,r,x);
        }
        else {
            write(query(l,r));
            putchar('\n');
        }
    }
    return 0;
}

namespace SegmentTree {
#define lc (a<<1)
#define rc ((a<<1)|1)
    int val[maxn<<2],tag[maxn<<2];
    inline void pushup(const int &a) {
        val[a]=val[lc]+val[rc];
    }
    inline void pushdown(const int &a,const int &l,const int &r,const int &mid) {
        val[lc]=tag[a]*(mid-l+1);
        val[rc]=tag[a]*(r-mid);
        tag[lc]=tag[rc]=tag[a];
        tag[a]=-1;
    }
    void Build(const int &a=1,const int &l=1,const int &r=n) {
        tag[a]=-1;
        if(l==r) return void(val[a]=b[l]);
        int mid=(l+r)>>1;
        Build(lc,l,mid);
        Build(rc,mid+1,r);
        pushup(a);
    }
    void Modify(const int &left,const int &right,const int &k,const int &a=1,const int &l=1,const int &r=n) {
        if(left>r||right<l) return;
        if(left<=l&&right>=r) {
            val[a]=(r-l+1)*k;
            tag[a]=k;
            return;
        }
        int mid=(l+r)>>1;
        if(~tag[a]) pushdown(a,l,r,mid);
        if(left<=mid) Modify(left,right,k,lc,l,mid);
        if(right>mid) Modify(left,right,k,rc,mid+1,r);
        pushup(a);
    }
    int QuerySection(const int &left,const int &right,const int &a=1,const int &l=1,const int &r=n) {
        if(left<=l&&right>=r) return val[a];
        int mid=(l+r)>>1,ans=0;
        if(~tag[a]) pushdown(a,l,r,mid);
        if(left<=mid) ans+=QuerySection(left,right,lc,l,mid);
        if(right>mid) ans+=QuerySection(left,right,rc,mid+1,r);
        return ans;
    }
#undef lc
#undef rc
}

如题，给定 nnn 个整数构成的序列 aaa，将对于指定的闭区间 [l,r][l, r][l,r] 查询其区间内的第 kkk 小值。

输入格式

第二行包含 nnn 个整数，第 iii 个整数表示序列的第 iii 个元素 aia_iai​。
接下来 mmm 行每行包含三个整数 l,r,kl, r, kl,r,k , 表示查询区间 [l,r][l, r][l,r] 内的第 kkk 小值。

对于每次询问，输出一行一个整数表示答案。

const int maxn=2e5+50;
int n,m,t;
int a[maxn],b[maxn];
struct node {
    int sum;
    node *lc,*rc;
}*root[maxn];

#define mid ((l+r)>>1)
node *build(int l,int r) {
    node *a=new node;
    a->sum=0;
    if(l<r) {
        a->lc=build(l,mid);
        a->rc=build(mid+1,r);
    }
    return a;
}
node *insert(node *pre,int l,int r,const int k) {
    node *a=new node;
    a->lc=pre->lc,a->rc=pre->rc,a->sum=pre->sum+1;
    if(l<r) {
        if(k<=mid) a->lc=insert(pre->lc,l,mid,k);
        else a->rc=insert(pre->rc,mid+1,r,k);
    }
    return a;
}
int query(node *u,node *v,int l,int r,const int k) {
    if(l>=r) return l;
    int x=v->lc->sum-u->lc->sum;
    if(k<=x) return query(u->lc,v->lc,l,mid,k);
    else return query(u->rc,v->rc,mid+1,r,k-x);
}

int main(void) {
    read(n),read(t);
    for(int i=1;i<=n;i++) {
        read(a[i]);
        b[i]=a[i];
    }
    std::sort(b+1,b+1+n);
    m=std::unique(b+1,b+1+n)-b-1;
    root[0]=build(1,m);
    for(int i=1;i<=n;i++) {
        int x=std::lower_bound(b+1,b+1+m,a[i])-b;
        root[i]=insert(root[i-1],1,m,x);
    }
    while(t--) {
        int l,r,k;
        read(l),read(r),read(k);
        write(b[query(root[l-1],root[r],1,m,k)]);
        putchar('\n');
    }
    return 0;
}

const int maxn=1e5+50;
int n,stamp,root;
int key[maxn],amount[maxn],size[maxn],pre[maxn],c[maxn][2];
inline void clear(int a) {
    key[a]=amount[a]=size[a]=pre[a]=c[a][0]=c[a][1]=0;
}
inline void gnew(int a,int ke,int fa) {
    key[a]=ke;
    amount[a]=size[a]=1;
    pre[a]=fa;
    c[a][0]=c[a][1]=0;
}
inline bool dir(int a) {
    return c[pre[a]][1]==a;
}//判断儿子类型
inline void replace(int a,int b,bool typ) {
    if(a) pre[a]=b;
    if(b) c[b][typ]=a;
}//用a作为b的typ儿子
inline void maintain(int a) {
    size[a]=amount[a]+size[c[a][0]]+size[c[a][1]];
}//维护信息
inline void rot(int a) {
    int f=pre[a],ff=pre[f];
    bool t=dir(a),tt=dir(f);
    int s=c[a][!t];
    replace(s,f,t);
    replace(a,ff,tt);
    replace(f,a,!t);
    maintain(f);
    maintain(a);
}//旋转
inline void splay(int a) {
    for(;pre[a];rot(a))
        if(pre[pre[a]]&&dir(a)==dir(pre[a]))
            rot(pre[a]);
    root=a;
}//把节点a旋到root
void insert(const int k) {
    if(!root) {
        gnew(++stamp,k,0);
        root=stamp;
        return;
    }
    int a=root,f=0;
    while(true) {
        if(k==key[a]) {
            amount[a]++;
            maintain(a);
            maintain(f);
            splay(a);
            return;
        }
        f=a;
        a=c[f][key[f]<k];
        if(!a) {
            gnew(++stamp,k,f);
            c[f][key[f]<k]=stamp;
            maintain(f);
            splay(stamp);
            return;
        }
    }
}//插入并旋转到根节点
int search(const int k) {
    int a=root,ans=0;
    while(true) {
        if(k<key[a]) a=c[a][0];
        else {
            if(c[a][0]) ans+=size[c[a][0]];
            if(k==key[a]) {
                splay(a);
                return ans+1;
            }
            ans+=amount[a];
            a=c[a][1];
        }
    }
}//查询k的排名并把k旋到根
int precursor() {
    int a=c[root][0];
    while(c[a][1]) a=c[a][1];
    return a;
}//根的前驱
int successor() {
    int a=c[root][1];
    while(c[a][0]) a=c[a][0];
    return a;
}//根的后继
void erase(const int k) {
    search(k);
    if(amount[root]>1) {
        amount[root]--;
        maintain(root);
    }
    else if(!c[root][0]&&!c[root][1]) {
        clear(root);
        root=0;
    }
    else if(!c[root][0]) {
        int oldroot=root;
        root=c[root][1];
        pre[root]=0;
        clear(oldroot);
    }
    else if(!c[root][1]) {
        int oldroot=root;
        root=c[root][0];
        pre[root]=0;
        clear(oldroot);
    }
    else {
        int p=precursor(),oldroot=root;
        splay(p);
        c[root][1]=c[oldroot][1];

        clear(oldroot);

    }

    int a=root;

        if(c[a][0]&&size[c[a][0]]>=k) a=c[a][0];
        else {
            int tmp=amount[a];
            if(c[a][0]) tmp+=size[c[a][0]];
            if(k<=tmp) return key[a];
            k-=tmp;
            a=c[a][1];
        }
    }
}//查找第k大的元素

可持久化平衡树

题目描述

您需要写一种数据结构（可参考题目标题），来维护一个可重整数集合，其中需要提供以下操作（ 对于各个以往的历史版本 ）：

1、 插入 xxx

2、 删除 xxx（若有多个相同的数，应只删除一个，如果没有请忽略该操作）

3、 查询 xxx 的排名（排名定义为比当前数小的数的个数 +1+1+1）

4、查询排名为 xxx 的数

5、 求 xxx 的前驱（前驱定义为小于 xxx，且最大的数，如不存在输出 −231+1-2^{31}+1−231+1）

6、求 xxx 的后继（后继定义为大于 xxx，且最小的数，如不存在输出 231−12^{31}-1231−1 ）

和原本平衡树不同的一点是，每一次的任何操作都是基于某一个历史版本，同时生成一个新的版本。（操作3, 4, 5, 6即保持原版本无变化）

每个版本的编号即为操作的序号（版本0即为初始状态，空树）

输入格式

第一行包含一个正整数 nnn ，表示操作的总数。

接下来 nnn 行，每行包含三个整数，第 iii 行记为 vi,opti,xiv_i, \text{opt}_i, x_ivi​,opti​,xi​。

viv_ivi​ 表示基于的过去版本号，opti\text{opt}_iopti​ 表示操作的序号， xix_ixi​ 表示参与操作的数值

        if(!a) a=++Cnt_Node;

        for(int i(1);i<=cnt_minus;i++) minus[i]=rc[minus[i]];
    }

    int Binary(int k,int l=1,int r=cnt) {
        int mid,s;
        while(l!=r) {
            mid=(l+r)>>1;
            s=GetSum();

namespace LinkCutTree {
#define lc c[x][0]
#define rc c[x][1]
    int c[maxn][2],val[maxn],sum[maxn],fa[maxn];
    char tag[maxn];

    inline bool dir(const int &x) {
        return x==c[fa[x]][1];
    }
    inline char notroot(const int &x) {
        return (x==c[fa[x]][0]||x==c[fa[x]][1])?0xff:0x00;
    }
    inline void pushup(const int &x) {
        sum[x]=sum[lc]^sum[rc]^val[x];
    }
    inline void pushtag(const int &x) {
        swap(lc,rc);
        tag[x]^=0xff;
    }
    inline void pushdown(const int &x) {
        if(!~tag[x]) {
            pushtag(lc);
            pushtag(rc);
            tag[x]=0x00;
        }
    }

    inline void connect(const int &x,const int &f,const int &d,const char &notr) {
        if(!~notr) c[f][d]=x;
        fa[x]=f;
    }
    inline void rotate(const int &x) {
        int f=fa[x],ff=fa[f],d=dir(x),t=dir(f);
        char p=notroot(f);
        connect(c[x][d^1],f,d,0xff);
        connect(f,x,d^1,0xff);
        connect(x,ff,t,p);
        pushup(f);
    }

    void pushin(const int &x) {
        if(!~notroot(x)) pushin(fa[x]);
        pushdown(x);
    }
    inline void Splay(const int &x) {
        pushin(x);
        for(int y(fa[x]);!~notroot(x);rotate(x),y=fa[x])
            if(!~notroot(y)) rotate((dir(x)^dir(y))?x:y);
        pushup(x);
    }

    inline void Access(int x) {
        for(int y(0);x;x=fa[y=x])
            Splay(x),rc=y,pushup(x);
    }
    inline void MakeRoot(const int &x) {
        Access(x);
        Splay(x);
        pushtag(x);
    }
    inline int FindRoot(int x) {
        Access(x);
        Splay(x);
        for(;lc;x=lc) pushdown(x);
        Splay(x);
        return x;
    }
    inline void Split(const int &x,const int &y) {
        MakeRoot(x);
        Access(y);
        Splay(y);
    }

    inline char Link(const int &x,const int &y) {
        MakeRoot(x);
        if(FindRoot(y)==x) return 0x00;
        fa[x]=y;
        return 0xff;
    }
    inline char Cut(const int &x,const int &y) {
        MakeRoot(x);
        if(FindRoot(y)!=x||fa[y]!=x||c[y][0]) return 0x00;
        fa[y]=lc=0;
        pushup(y);
        return 0xff;
    }

}
