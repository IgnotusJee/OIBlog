const int maxn=2e4+50,maxm=1e6+50,inf=0x3f3f3f3f;
int n,m,k,a[maxn],b[maxn];

int suf[maxn],rank[maxn],tmp[maxn],tax[maxn],height[maxn];

inline void BaseSort() {
    memset(tax,0,sizeof tax);
    for(R i(1);i<=n;i++) tax[rank[i]]++;
    for(R i(1);i<=m;i++) tax[i]+=tax[i-1];
    for(R i(n);i;i--) suf[ tax[rank[tmp[i]]]-- ]=tmp[i];
}

void SuffixSort() {
    m='z'-'0'+1;
    for(R i(1);i<=n;i++) rank[i]=s[i]-'0'+1,tmp[i]=i;
    BaseSort();
    for(R p(0),w(1);p<n;m=p,w<<=1) {
        p=0;
        for(R i(1);i<=w;i++) tmp[++p]=n-w+i;
        for(R i(1);i<=n;i++) if(suf[i]>w) tmp[++p]=suf[i]-w;
        BaseSort();
        std::swap(tmp,rank);
        rank[suf[1]]=p=1;
        for(R i(2);i<=n;i++)
            rank[suf[i]]=(tmp[suf[i]]==tmp[suf[i-1]]&&tmp[suf[i]+w]==tmp[suf[i-1]+w]) ? p : ++p;
    }
}

void GetHight() {
    for(R i(1),k(0);i<=n;i++) {
        if(k) k--;
        while(a[i+k]==a[suf[rank[i]-1]+k]) k++;
        height[rank[i]]=k;
    }
}

SAMSAMSAM模板

namespace SAM {
    struct Node {
        int link,len,next[26];
    }st[maxn<<1];
    int Node_Cnt,last;
    void Init() {st[Node_Cnt++].link=-1;}

    inline void Insert(const int &c) {
        int cur(Node_Cnt++);
        st[cur].len=st[last].len+1;
        int p(last);
        for(;~p&&!st[p].next[c];p=st[p].link) st[p].next[c]=cur;
        if(~p) {
            int q(st[p].next[c]);
            if(st[p].len+1==st[q].len) st[cur].link=q;
            else {
                int clone(Node_Cnt++);st[clone]=st[q];
                st[clone].len=st[p].len+1;
                for(;~p&&st[p].next[c]==q;p=st[p].link) st[p].next[c]=clone;
                st[cur].link=st[q].link=clone;
            }
        }
        last=cur;
    }
}
