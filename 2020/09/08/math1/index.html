一. 组合数公式

Cmn=m!n!(m−n)!,n≥mC^n_m = \frac{m!}{n!(m-n)!} , n \geq m Cmn​=n!(m−n)!m!​,n≥m

	C[i][0]=1;

inline int euler_varphi(int n) {
	int ans=n,m=sqrt(n);
	for(R i=2;i<=m;i++) {
		if(n%i) continue;
		ans=ans/i*(i-1);
		while(n%i==0) n/=i;
	}
	if(n>1) ans=ans/n*(n-1);
	return ans;
}
