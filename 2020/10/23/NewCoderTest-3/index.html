T1 牛半仙的妹子数

降智题。打了一个小时的表毛都没想出来，但其实结果非常简单。

考虑两种操作都不会使A+B+CA+B+CA+B+C改变。

当A+B≤CA+B\le CA+B≤C时，C′=C−A−B=2C−(A+B+C)=2C mod (A+B+C)C ' = C-A-B= 2C - (A+B+C) = 2C \bmod {(A+B+C)}C′=C−A−B=2C−(A+B+C)=2Cmod(A+B+C)；

当A+B>CA+B>CA+B>C时，C′=2C=2C mod (A+B+C)C' = 2C = 2C \bmod{(A+B+C)}C′=2C=2Cmod(A+B+C)；

综上，变化kkk次后C=C×2k mod (A+B+C)C=C\times 2^k \bmod (A+B+C)C=C×2kmod(A+B+C)。快速幂即可。

using ll = long long ;

inline ll QuickPow(ll a,ll b,const ll &p) {
	ll ans(1);
	for(;b;b>>=1,a=a*a%p) if(b&1) ans=ans*a%p;
	return ans;
}

int main(void) {
	int T;cin>>T;
	ll a,b,c,k;
	while(T--&&cin>>a>>b>>c>>k) cout<<c*QuickPow(2,k,a+b+c)%(a+b+c)<<endl;
	return 0;
}

T2 牛半仙的妹子图

sb题。考虑只有最小瓶颈生成树上的边才能产生贡献，而最小生成树就是一棵最小瓶颈生成树。在Kruscal求最小生成树时在并查集中加入bitset维护颜色，记录每条边加入后的颜色数并维护前缀和。查询时在排序过的边权数组上二分，类似分块的方法查询。

注意起点也有颜色。我怎么觉得是题意表述不清，为什么自己家里还有妹子

const int maxn(5e5+10),maxc(600+10);
int n,m,q,s,c[maxn],opt;
using ll = long long;
ll mod;

struct Edge {
	int u,v,w;
}e[maxn];

class UnionAndFindSet {
private:
	int fa[maxn],rank[maxn];
	std::bitset<maxc> b[maxn];
	int Find(const int &x) {
		return x == fa[x] ? x : fa[x] = Find(fa[x]) ;
	}
public:
	void Init() {
		for(int i(1);i<=n;i++) {
			fa[i]=i;rank[i]=1;b[i]=0ULL;b[i].set(c[i]);
		}
	}
	inline void Merge(const int &x,const int &y) {
		int u(Find(x)),v(Find(y));
		if(u==v) return;
		if(rank[u]<rank[v]) swap(u,v);
		fa[v]=u;rank[u]=max(rank[u],rank[v]+1);
		b[u]|=b[v];
	}
	inline bool Same(const int &u,const int &v) {
		return Find(u)==Find(v);
	}
	inline int Color(const int &u) {
		return b[Find(u)].count();
	}
}U;

int cnt;ll val[maxn],ans[maxn],sum[maxn];
void Kruscal() {
	std::sort(e+1,e+1+m,[](const Edge &p,const Edge &q) {
		return p.w<q.w;
	});
	U.Init();cnt=1;ans[1]=1;
	for(int i(1);i<=m&&cnt<n;i++) {
		if(U.Same(e[i].u,e[i].v)) continue;
		U.Merge(e[i].u,e[i].v);
		val[++cnt]=e[i].w;
		ans[cnt]=U.Color(s);
	}
	for(int i(1);i<cnt;i++)
		sum[i]=sum[i-1]+ans[i]*ll(val[i+1]-val[i]);
}

inline ll Query(const ll &l,const ll &r) {
	int p(std::upper_bound(val+1,val+cnt+1,l)-val-1); // 第一个小于等于l的
	int q(std::upper_bound(val+1,val+cnt+1,r)-val-1); // 第一个小于等于r的
	if(p==q) return (r-l+1)*ans[p];
	ll Ans(sum[q-1]-sum[p]);
	Ans+=(val[p+1]-l)*ans[p];
	Ans+=(r-val[q]+1)*ans[q];
	return Ans;
}

int main(void) {
	cin>>n>>m>>q>>s>>opt;
	if(opt) cin>>mod;
	for(int i(1);i<=n;i++) cin>>c[i];
	for(int i(1);i<=m;i++) cin>>e[i].u>>e[i].v>>e[i].w;
	Kruscal();ll last(0LL);
	for(int i(1);i<=q;i++) {
		ll l,r;cin>>l>>r;
		if(opt) {
			l=(l^last)%mod+1;
			r=(r^last)%mod+1;
		}
		if(l>r) swap(l,r);
		cout<<(last=Query(l,r))<<endl;
	}
	return 0;
}

T3 牛半仙的妹子Tree

神题。每次询问即对于点u和当前时间t，是否能找出一个点v，设其染色时间为tvt_vtv​，使得dist(u,v)≤t−tvdist(u,v)\le t - t_vdist(u,v)≤t−tv​。

拆解式子

dist(u,v)≤t−tvdist(1,u)+dist(1,v)−2×dist(1,lcau,v)≤t−tvdist(1,v)+tv−2×dist(1,lcau,v)≤t−dist(1,u)dist(u,v)\le t-t_v \\ dist(1,u) + dist(1,v) - 2\times dist(1,lca_{u,v})\le t - t_v \\ dist(1,v)+t_v- 2\times dist(1,lca_{u,v})\le t - dist(1,u) \\ dist(u,v)≤t−tv​dist(1,u)+dist(1,v)−2×dist(1,lcau,v​)≤t−tv​dist(1,v)+tv​−2×dist(1,lcau,v​)≤t−dist(1,u)

上式为充分必要条件，弱化为充分不必要条件为

dist(1,v)+tv−2×dist(1,f)≤t−dist(1,u)dist(1,v) + t_v - 2 \times dist(1,f) \le t -dist(1,u) dist(1,v)+tv​−2×dist(1,f)≤t−dist(1,u)

取fff为u,vu,vu,v的公共祖先，则当且仅当f=lcau,vf = lca_{u,v}f=lcau,v​时式子为必要，此时dist(1,f)dist(1,f)dist(1,f)取得最大值。

对于uuu的染色操作，只对其到根的路径有影响。

树剖，区间维护上式的最小值。为了保证充分必要性，需要额外维护区间dist(1,f)dist(1,f)dist(1,f)的最大值，查询时用它来计算维护最小值。此时，查询某个点到根的路径，就可以查到所有可能成为答案的式子。

为了简单，线段树可以标记永久化。

const int maxn(1e5+10),inf(0x3f3f3f3f);
int n,m;

namespace Tree {
	struct Edge {
		int v,next;
		Edge(const int &v,const int &next) :v(v),next(next) {}
		Edge() {}
	}e[maxn<<1];
	int head[maxn],cnt;
	inline void AddEdge(const int &u,const int &v) {
		e[++cnt]=Edge(v,head[u]);
		head[u]=cnt;
	}

	int size[maxn],fa[maxn],son[maxn],dist[maxn];
	void Dfs1(const int &u=1) {
		size[u]=1;
		for(int i(head[u]);i;i=e[i].next) {
			const int &v(e[i].v);
			if(v==fa[u]) continue;
			fa[v]=u;dist[v]=dist[u]+1;
			Dfs1(v);
			size[u]+=size[v];
			if(size[v]>size[son[u]]) son[u]=v;
		}
	}
	int dfn[maxn],stamp,top[maxn],pos[maxn];
	void Dfs2(const int &u=1,const int &anc=1) {
		dfn[u]=++stamp;pos[stamp]=u;top[u]=anc;
		if(son[u]) Dfs2(son[u],anc);
		for(int i(head[u]);i;i=e[i].next) {
			const int &v(e[i].v);
			if(v==fa[u]||v==son[u]) continue;
			Dfs2(v,v);
		}
	}

	namespace SegmentTree {
	#define lc (a<<1)
	#define rc ((a<<1)|1)
		int val[maxn<<2],minv[maxn<<2],tag[maxn<<2];
		// 分别为初始最大值 , 最小值即答案 , 区间取最小值永久化标记(没有减掉初始最大值)
		void Build(const int &a=1,const int &l=1,const int &r=n) {
			minv[a]=tag[a]=inf;
			if(l==r) return void(val[a]=dist[pos[l]]);
			int mid((l+r)>>1);
			Build(lc,l,mid);
			Build(rc,mid+1,r);
			val[a]=max(val[lc],val[rc]);
		}

		char clr[maxn<<2];
		inline void pushup(const int &a) {
			minv[a]=min(tag[a]-2*val[a],minv[lc],minv[rc]);
		}
		inline void pushclear(const int &a=1) {
			clr[a]=0xff;tag[a]=minv[a]=inf;
		}
		inline void pushdown(const int &a) {
			if(!~clr[a]) {
				pushclear(lc);pushclear(rc);
				clr[a]=0x00;
			}
		}

		void Modify(const int &left,const int &right,const int &k,const int &a=1,const int &l=1,const int &r=n) {
			if(left<=l&&right>=r) return void(minv[a]=min(minv[a],(tag[a]=min(tag[a],k))-2*val[a]));
			pushdown(a);
			int mid((l+r)>>1);
			if(left<=mid) Modify(left,right,k,lc,l,mid);
			if(right>mid) Modify(left,right,k,rc,mid+1,r);
			pushup(a);
		}
		int Query(const int &left,const int &right,const int &k=inf,const int &a=1,const int &l=1,const int &r=n) {
			if(left<=l&&right>=r) return min(minv[a],k-2*val[a]);
			pushdown(a);
			int mid((l+r)>>1);
			if(right<=mid) return Query(left,right,min(k,tag[a]),lc,l,mid);
			if(left>mid) return Query(left,right,min(k,tag[a]),rc,mid+1,r);
			return min(Query(left,right,min(k,tag[a]),lc,l,mid),Query(left,right,min(k,tag[a]),rc,mid+1,r));
		}
	#undef lc
	#undef rc
	}
	using SegmentTree::pushclear;

	inline void Modify(int u,const int &t) {
		const int k(dist[u]+t);
		for(;u;u=fa[top[u]]) SegmentTree::Modify(dfn[top[u]],dfn[u],k);
	}
	inline int Query(int u) {
		int ans(inf);
		for(;u;u=fa[top[u]]) ans=min(ans,SegmentTree::Query(dfn[top[u]],dfn[u]));
		return ans;
	}
}

int main(void) {
	cin>>n>>m;
	for(int i(1);i<n;i++) {
		int u,v;cin>>u>>v;
		Tree::AddEdge(u,v);
		Tree::AddEdge(v,u);
	}
	Tree::Dfs1();Tree::Dfs2();Tree::SegmentTree::Build();
	for(int i(1);i<=m;i++) {
		int opt,x;
		cin>>opt>>x;
		if(opt==1) Tree::Modify(x,i);
		else if(opt==2) Tree::pushclear();
		else if(opt==3) cout<<(Tree::Query(x)<=i-Tree::dist[x]?"wrxcsd":"orzFsYo")<<endl;
	}
	return 0;
}

T4 牛半仙的魔塔（增强版）

鬼题。

首先有几个显然的结论：

  • 勇士的攻击不会变，于是对于打每个怪物的回合是不变的。
  • 任何时候勇士的防不会大于怪物的攻击，所以每单位防的效果对每个怪物是相同的。

所以我们要安排顺序，使得减小的伤害最大。

设怪物iii的回合数为hih_ihi​，增加的防御为bib_ibi​。考虑一个最优的打怪序列{p1,p2,…,pn}\{ p_1,p_2,\ldots,p_n \}{p1​,p2​,…,pn​}，若交换pi,pi+1p_i,p_{i+1}pi​,pi+1​，则交换后的减伤会减小。则有：bi+1×hi−bi×hi+1≤0b_{i+1}\times h_i - b_i \times h_{i+1} \le 0bi+1​×hi​−bi​×hi+1​≤0。

移项得bihi≥bi+1hi+1\frac{b_i}{h_i}\ge \frac{b_{i+1}}{h_{i+1}}hi​bi​​≥hi+1​bi+1​​。

设ti=bihit_{i}=\frac{b_i}{h_i}ti​=hi​bi​​，则按照tit_iti​降序排序即可。

但问题在树上，有些点必需要经过前置点才能打。考虑另一个结论：对于一个点iii和它的后置点jjj，若有ti<tjt_i<t_jti​<tj​，则打完iii后马上打jjj一定最优。证明根据前面的结论显然。

于是对于每个满足条件的点集，设其回合数为其元素回合数之和，增加的防御也是其元素增加的防御之和。从起点开始，将每个当前新到达的点加入大根堆中即可。复杂度nlog(n)nlog(n)nlog(n)。

const int maxn(1e5+10);
int n;
using ll = long long ;

struct Edge {
	int v,next;
	Edge(const int &v,const int &next) :v(v),next(next) {}
	Edge() {}
}e[maxn<<1];
int head[maxn],cnt;
inline void AddEdge(const int &u,const int &v) {
	e[++cnt]=Edge(v,head[u]);
	head[u]=cnt;
}
int fa[maxn];

ll hp[maxn],attack[maxn],defence[maxn],val[maxn],round_[maxn];
ll Round[maxn],Val[maxn];

struct UAFS {
	int fa[maxn];
	void Init() {for(int i(1);i<=n;i++) fa[i]=i;}
	int find(const int &x) {return x == fa[x] ? x : fa[x] = find(fa[x]);}
}U;

void Dfs1(const int &u=1) {
	for(int i(head[u]);i;i=e[i].next) {
		const int &v(e[i].v);
		if(v==fa[u]) continue;
		fa[v]=u;Dfs1(v);
	}
}

char access[maxn],vis[maxn];
std::vector<int> g[maxn];
void Dfs2(const int &u)  {
	hp[1]-=round_[u]*(attack[u]-defence[1]);
	defence[1]+=val[u];
	access[u]=0xff;
	for(const auto &v: g[u]) Dfs2(v);
}

int main(void) {
	using pii=std::pair<double,int>;
	static std::priority_queue<pii> q;
	cin>>n;
	for(int i(1);i<n;i++) {
		int u,v;cin>>u>>v;
		AddEdge(u,v);AddEdge(v,u);
	}
	Dfs1();
	cin>>hp[1]>>attack[1]>>defence[1];
	for(int i(2);i<=n;i++) {
		cin>>hp[i]>>attack[i]>>defence[i]>>val[i];
		Round[i]=round_[i]=hp[i]/(attack[1]-defence[i])-(hp[i]%(attack[1]-defence[i])==0);
		Val[i]=val[i];q.emplace(double(val[i])/double(round_[i]),i);
	}
	U.Init();
	access[1]=0xff;
	while(!q.empty()) {
		int u(q.top().second);q.pop();
		if(!~vis[u]) continue;
		vis[u]=0xff;
		if(!~access[fa[u]]) {Dfs2(u);continue;}
		int &&v(U.find(fa[u]));
		Round[v]+=Round[u];
		Val[v]+=Val[u];
		g[v].push_back(u);
		U.fa[u]=v;
		q.emplace(double(Val[v])/double(Round[v]),v);
	}
	cout<<hp[1]<<endl;
	return 0;
}
