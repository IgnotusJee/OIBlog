2020-10-13考试总结

T1 破冰派对

考试时一看就觉得要求出最大团，于是花了一个多小时抠这个。于是最终也没有抠出来，输出大样例跑路了。考完了才知道最大团问题是NP完全的。

其实正解就是暴搜。因为状态数不多，所以搜的出来。

教训：一段时间内想不出正解考虑打暴力跑路。

const int maxn(1000+10);
int n,m;
char map[maxn][maxn];

int work[maxn],won,part[maxn],pan;
int ans;
void Dfs(const int &u) {
	if(u==n+1) return void(ans+=(won&&pan));
	char flag(0x00);
	for(int i(1);i<=won;i++)
		if(~map[u][work[i]]) {
			flag=0xff;
			break;
		}
	if(~flag) {
		work[++won]=u;
		Dfs(u+1);
		won--;
	}
	flag=0x00;
	for(int i(1);i<=pan;i++)
		if(!~map[u][part[i]]) {
			flag=0xff;
			break;
		}
	if(~flag) {
		part[++pan]=u;
		Dfs(u+1);
		pan--;
	}
}

int main(void) {
#ifndef ONLINE_JUDGE
	freopen("party.in","r",stdin);
#endif
	int T;cin>>T;
	while(T--) {
		ans=0;
		memset(map,0x00,sizeof map);
		cin>>n>>m;
		for(int i(1);i<=m;i++) {
			int u,v;cin>>u>>v;
			map[u][v]=map[v][u]=0xff;
		}
		Dfs(1);
		cout<<ans<<endl;
	}
	return 0;
}

T2 无聊游戏

这题还是考试时想出来了。考虑改变前缀使得后缀最大和最小。

// 小A想让数列和尽量小，则他要选择最大后缀变负
// 小N要让最大后缀最小，同时修改的前缀和为负且大
// 从左向右扫，维护左端点到当前点的最大后缀，这个区间要乘上-A。
// 预处理出后缀和和最大后缀
const int maxn(1e6+10);
int n;
using ll = long long;
ll A,B,q[maxn];
ll sumpre[maxn];
ll sumsuf[maxn],lensuf[maxn],possuf[maxn];

inline ll Calc(const int &i,const int &j) {
	if(i<j) return sumpre[i]*-A+sumsuf[j]*-B+sumpre[j-1]-sumpre[i];
	else return sumpre[j-1]*-A+sumsuf[i+1]*-B+(sumpre[i]-sumpre[j-1])*A*B;

}

int main(void) {
#ifndef ONLINE_JUDGE
	freopen("game.in","r",stdin);
//	freopen("game.out","w",stdout);
#endif
	cin>>n>>A>>B;
	for(int i(1);i<=n;i++) cin>>q[i],sumpre[i]=sumpre[i-1]+q[i];
	possuf[n+1]=n+1;
	for(int i(n);i;i--) {
		sumsuf[i]=q[i]+sumsuf[i+1];
		if(sumsuf[i]>lensuf[i+1]) {
			lensuf[i]=sumsuf[i];
			possuf[i]=i;
		}
		else {
			lensuf[i]=lensuf[i+1];
			possuf[i]=possuf[i+1];
		}
	}
	ll ans(Calc(0,possuf[1])),sum(0);
	for(int i(1),t(1);i<=n;i++) {// t为前缀的最大后缀位置
		if(sum+q[i]*(-A)<0) sum=0,t=i+1;
		else sum+=q[i]*(-A);
		int nowpos;
		if(lensuf[i+1]>sum+sumsuf[i+1]) nowpos=possuf[i+1];
		else nowpos=t;
		ans=max(ans,Calc(i,nowpos));
	}
	cout<<ans<<endl;
	return 0;
}

T3 建设国家

又是最生成树。考试时根本没有思路，因为快没有时间了就乱打了个贪心。其实也不难想出来。

考虑第一轮计划对每一条边的贡献就是题目中给出的那个式子。第二轮计划拆成一条边上两点加人对对方的影响。推式子后发现优先加F大的，权值转化到边权上做最小生成树即可。

const int maxn(55),maxm(maxn*maxn);
using ll=long long;
int n,m;ll T[maxn],F[maxn],U[maxn],C;
char map[maxn][maxn];
struct Edge {
	int u,v;ll w;
	Edge(const int &u,const int &v,const ll &w) :u(u),v(v),w(w) {}
	Edge() {}
}e[maxm];
inline ll Calc(const int &i,const int &j) {
	if(F[i]<F[j]) return Calc(j,i);
	return 1ll*F[i]*(U[i]-T[i])*T[j]+1ll*F[j]*(U[j]-T[j])*U[i];
}

class UnionAndFindSet {
private:
	int fa[maxn],rank[maxn];
	int Find(const int &x) {
		return x == fa[x] ? x : fa[x] = Find(fa[x]);
	}
public:
	inline void Init() {
		for(int i(1);i<=n;i++) fa[i]=i,rank[i]=1;
	}
	inline void Merge(const int &x,const int &y) {
		int u(Find(x)),v(Find(y));
		if(u==v) return;
		if(rank[u]<rank[v]) swap(u,v);
		fa[v]=u;rank[u]=max(rank[u],rank[v]+1);
	}
	inline bool Same(const int &x,const int &y) {
		return Find(x)==Find(y);
	}
}S;

int main(void) {
#ifndef ONLINE_JUDGE
	freopen("res.in","r",stdin);
#endif
	cin>>n;
	S.Init();ll ans(0);
	for(int i(1);i<=n;i++) cin>>T[i];
	for(int i(1);i<=n;i++) cin>>U[i];
	for(int i(1);i<=n;i++) cin>>F[i],ans+=(T[i]+U[i]-1)*(U[i]-T[i])/2*F[i];
	for(int i(1);i<=n;i++) cin>>&map[i][1];
	cin>>C;
	for(int i(1);i<=n;i++)
		for(int j(1);j<i;j++) {
			if(map[i][j]=='Y') {
				ans+=Calc(i,j);
				S.Merge(i,j);
			}
			else e[++m]=Edge(i,j,Calc(i,j)+(T[i]+T[j])*C);
		}
	std::sort(e+1,e+1+m,[](const Edge &p,const Edge &q) {
		return p.w<q.w;
	});
	for(int i(1);i<=m;i++) {
		if(S.Same(e[i].u,e[i].v)) continue;
		S.Merge(e[i].u,e[i].v);
		ans+=e[i].w;
	}
	cout<<ans<<endl;
	return 0;
}
