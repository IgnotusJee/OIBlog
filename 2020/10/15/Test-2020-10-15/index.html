T1 运货

考试时想了半个小时想出了正解。但是由于鬼畜的数据范围就当作暴力了没优化空间，于是成功没过。

按照出现时间减去到达时间排序，一个一个选就行了。

const int maxn(1e6+10);
typedef long long ll;
int n,m;
int a[maxn];
ll s[maxn*10],tot;

int main(void) {
#ifndef ONLINE_JUDGE
	freopen("Shipment.in","r",stdin);
	freopen("Shipment.out","w",stdout);
#endif
	cin>>n>>m;
	for(int i(1);i<=n;i++) {
		cin>>a[i];
		for(int j(1);j<=a[i];j++) {
			ll x;cin>>x;
			s[++tot]=x-ll(i);
		}
	}
	std::sort(s+1,s+1+tot);
	static std::priority_queue<ll,std::vector<ll>,std::greater<ll>> q;//小根堆
	for(int i(1);i<=m;i++) q.push(0);
	ll ans(0);
	for(int i(1);i<=tot;i++) {
		ll k(q.top());q.pop();
		ll now(max(s[i]+ll(n)+1,k+ll(n)+1));
		ans=max(ans,now);//什么时候出现的，加上到右边的
		q.push(now);
	}
	cout<<ans<<endl;
	return 0;
}

T2 维护

考试时觉得可能是LCT，但LCT不熟并且耗时，所以选择了打指针暴力。可惜出锅了。

正解是LCT，每个点没被删除权值就为1，否则为0。每次在一个链的Splay上二分查找就行了。

const int maxn(2e5+10);
int q,online;

namespace LCT {
#define lc c[x][0]
#define rc c[x][1]
	int size[maxn],val[maxn],c[maxn][2],fa[maxn];
	char tag[maxn];

	inline int dir(const int &x) {
		return x==c[fa[x]][1];
	}
	inline char notroot(const int &x) {
		return ~char(!(x==c[fa[x]][0]||x==c[fa[x]][1]));
	}
	inline void pushup(const int &x) {
		size[x]=size[lc]+size[rc]+val[x];
	}
	inline void pushtag(const int &x) {
		tag[x]^=0xff;
		swap(lc,rc);
	}
	inline void pushdown(const int &x) {
		if(!~tag[x]) {
			pushtag(lc);
			pushtag(rc);
			tag[x]=0x00;
		}
	}

	inline void connect(const int &x,const int &f,const int &d,const char &notr) {
		if(!~notr) c[f][d]=x;
		fa[x]=f;
	}
	inline void Rotate(const int &x) {
		int f(fa[x]),ff(fa[f]),d(dir(x)),t(dir(f));
		char p(notroot(f));
		connect(c[x][d^1],f,d,0xff);
		connect(f,x,d^1,0xff);
		connect(x,ff,t,p);
		pushup(f);
	}

	void pushstack(const int &x) {
		if(!~notroot(x)) pushstack(fa[x]);
		pushdown(x);
	}
	inline void Splay(const int &x) {
		pushstack(x);
		for(;!~notroot(x);Rotate(x))
			if(!~notroot(fa[x])) Rotate(dir(x)^dir(fa[x])?x:fa[x]);
		pushup(x);
	}

	inline void Access(int x) {
		for(int y(0);x;x=fa[y=x])
			Splay(x),rc=y,pushup(x);
	}
	inline void MakeRoot(const int &x) {
		Access(x);
		Splay(x);
		pushtag(x);
	}

	inline int Find(int x,int k) {
		if(!k) return x;
		Access(x);Splay(x);
		x=lc;
		while(x) {
			pushdown(x);
			if(k<=size[rc]) x=rc;
			else if(k==val[x]+size[rc]) return x;
			else k-=val[x]+size[rc],x=lc;
		}
		return 0;
	}

	inline void Insert(const int &x,const int &y) {
		fa[y]=x;val[y]=size[y]=1;
		Access(y);
	}
	inline int Remove(const int &x) {
		val[x]=0;
		return Find(x,1);
	}
#undef lc
#undef rc
}

char deled[maxn];
int main(void) {
#ifndef ONLINE_JUDGE
	freopen("maintain.in","r",stdin);
	freopen("maintain.out","w",stdout);
#endif
	cin>>q>>online;
	LCT::val[1]=LCT::size[1]=1;
	for(int i(1),tot(1),lastans(0),ans;i<=q;i++) {
		int opt,x,y;cin>>opt>>x;
		if(opt==1) {
			x^=lastans;
			if(x&&x<=tot&&~deled[x]) {
				LCT::Insert(x,++tot);
				if(online) lastans=x;
			}
		}
		else if(opt==2) {
			x^=lastans;
			if(x&&x<=tot&&~deled[x]) {
				ans=LCT::Remove(x);
				deled[x]=0xff;
				if(online) lastans=ans;
			}
		}
		else {
			cin>>y;x^=lastans;y^=lastans;
			if(x&&x<=tot&&~deled[x]) {
				cout<<(ans=LCT::Find(x,y))<<endl;
				if(online) lastans=ans;
			}
		}
	}
	return 0;
}

T3 相似

考试时打出来了，但是有一种情况没考虑，就是AAS且有唯一解的情况。通过边和角的sin值判断，可惜数据里根本没有没有唯一解的情况。

const double Pi(acos(-1.0));
int T,x,y;
double a,b,c,A,B,C,d,e,f,D,E,F;

inline void SAS(double &a,double &b,double &c,double &A,double &B,double &C) {
	b=sqrt(a*a+c*c-2*a*c*cos(B/180.0*Pi));
	A=asin(a*sin(B/180.0*Pi)/b)/Pi*180.0;
	C=asin(c*sin(B/180.0*Pi)/b)/Pi*180.0;
}//a , B , c

inline void SSS(double &a,double &b,double &c,double &A,double &B,double &C) {
	B=acos((a*a+c*c-b*b)/2/a/c)/Pi*180.0;
	A=asin(a*sin(B/180.0*Pi)/b)/Pi*180.0;
	C=asin(c*sin(B/180.0*Pi)/b)/Pi*180.0;
}// a, b, c

inline void AA(double &A,double &B,double &C) {
	B=180.0-A-C;
}// A , C

inline void SSA_(double &a,double &b,double &c,double &A,double &B,double &C) {
	if(!a) a=sqrt(b*b-c*c);
	if(!c) c=sqrt(b*b-a*a);
	if(!b) b=sqrt(a*a+c*c);
	A=asin(a/b)/Pi*180.0;C=asin(c/b)/Pi*180.0;
}// B = Pi/2

inline void SSA(double &a,double &b,double &c,double &A,double &B,double &C) {
	C=asin(c*sin(A/180.0*Pi)/a)/Pi*180.0;
	B=180.0-A-C;
	b=a/sin(a)*sin(b);
}// A , a , c

inline int cmp(const double &x,const double &y) {
	if(fabs(x-y)<1e-2) return 1;
	return 0;
}

inline void Solve() {
	char flag1(0x00),flag2(0x00);
	if(x==3) {
		int s;
		s=cmp(A,0)+cmp(B,0)+cmp(C,0);
		if(s==3) SSS(a,b,c,A,B,C),flag1=0xff;
		else if(s==1) {
			flag1=0xff;
			if(cmp(A,0)) AA(B,A,C);
			else if(cmp(B,0)) AA(A,B,C);
			else if(cmp(C,0)) AA(A,C,B);
		}
		else if(s==2) {
			if(!cmp(A,0)) {
				if(!cmp(b,0)&&!cmp(c,0)) {
					SAS(b,a,c,B,A,C);flag1=0xff;
				}
				else if(cmp(A,90.0)) SSA_(b,a,c,B,A,C),flag1=0xff;
				else {
					if(!cmp(b,0)) SSA(a,c,b,A,C,B),flag1=0xff;
					else if(!cmp(c,0)) SSA(a,b,c,A,B,C),flag1=0xff;
				}
			}
			else if(!cmp(B,0)) {
				if(!cmp(a,0)&&!cmp(c,0)) {
					SAS(a,b,c,A,B,C);flag1=0xff;
				}
				else if(cmp(B,90.0)) SSA_(a,b,c,A,B,C),flag1=0xff;
				else {
					if(!cmp(a,0)) SSA(b,c,a,B,C,A),flag1=0xff;
					else if(!cmp(c,0)) SSA(b,a,c,B,A,C),flag1=0xff;
				}
			}
			else if(!cmp(C,0)) {
				if(!cmp(a,0)&&!cmp(b,0)) {
					SAS(a,c,b,A,C,B);flag1=0xff;
				}
				else if(cmp(C,90.0)) SSA_(a,c,b,A,C,B),flag1=0xff;
				else {
					if(!cmp(a,0)) SSA(c,b,a,C,B,A),flag1=0xff;
					else if(!cmp(b,0)) SSA(c,a,b,C,A,B),flag1=0xff;
				}
			}
		}
		else SSS(a,b,c,A,B,C),flag1=0xff;
	}
	else if(x==2) {
		int s;
		s=cmp(A,0)+cmp(B,0)+cmp(C,0);
		if(s==1) {
			flag1=0xff;
			if(cmp(A,0)) AA(B,A,C);
			else if(cmp(B,0)) AA(A,B,C);
			else if(cmp(C,0)) AA(A,C,B);
		}
	}
	if(y==3) {
		int s;
		s=cmp(D,0)+cmp(E,0)+cmp(F,0);
		if(s==3) SSS(d,e,f,D,E,F),flag2=0xff;
		else if(s==1) {
			flag2=0xff;
			if(cmp(D,0)) AA(E,D,F);
			else if(cmp(E,0)) AA(D,E,F);
			else if(cmp(F,0)) AA(D,F,E);
		}
		else if(s==2) {
			if(!cmp(D,0)) {
				if(!cmp(e,0)&&!cmp(f,0)) {
					SAS(e,d,f,E,D,F);flag2=0xff;
				}
				else if(cmp(D,90.0)) SSA_(e,d,f,E,D,F),flag2=0xff;
				else {
					if(!cmp(e,0)) SSA(d,f,e,D,F,E),flag2=0xff;
					else if(!cmp(f,0)) SSA(d,e,f,D,E,F),flag2=0xff;
				}
			}
			else if(!cmp(E,0)) {
				if(!cmp(d,0)&&!cmp(f,0)) {
					SAS(d,e,f,D,E,F);flag2=0xff;
				}
				else if(cmp(E,90.0)) SSA_(d,e,f,D,E,F),flag2=0xff;
				else {
					if(!cmp(d,0)) SSA(e,f,d,E,F,D),flag2=0xff;
					else if(!cmp(f,0)) SSA(e,d,f,E,D,F),flag2=0xff;
				}
			}
			else if(!cmp(F,0)) {
				if(!cmp(d,0)&&!cmp(e,0)) {
					SAS(d,f,e,D,F,E);flag2=0xff;
				}
				else if(cmp(F,90.0)) SSA_(d,f,e,D,F,E),flag2=0xff;
				else {
					if(!cmp(d,0)) SSA(f,e,d,F,E,D),flag2=0xff;
					else if(!cmp(e,0)) SSA(f,d,e,F,D,E),flag2=0xff;
				}
			}
		}
		else SSS(d,e,f,D,E,F),flag2=0xff;
	}
	else if(y==2) {
		int s;
		s=cmp(D,0)+cmp(E,0)+cmp(F,0);
		if(s==1) {
			flag2=0xff;
			if(cmp(D,0)) AA(E,D,F);
			else if(cmp(E,0)) AA(D,E,F);
			else if(cmp(F,0)) AA(D,F,E);
		}
	}
	if(!~flag1&&!~flag2&&cmp(A,D)&&cmp(B,E)&&cmp(C,F)) cout<<"YES\n";
	else cout<<((!cmp(A,0)&&!cmp(D,0)&&cmp(A,D))
		+(!cmp(B,0)&&!cmp(E,0)&&cmp(B,E))
		+(!cmp(C,0)&&!cmp(F,0)&&cmp(C,F)))<<endl;
}

int main(void) {
#ifndef ONLINE_JUDGE
	freopen("similarity.in","r",stdin);
#endif
	cin>>T;
	while(T--) {
		a=b=c=A=B=C=d=e=f=D=E=F=0.0;
		cin>>x>>y;
		for(int i(1);i<=x+y;i++) {
			char ch;double xx;
			cin>>ch>>xx;
			switch(ch) {
				case 'a': a=xx;break;
				case 'b': b=xx;break;
				case 'c': c=xx;break;
				case 'A': A=xx;break;
				case 'B': B=xx;break;
				case 'C': C=xx;break;
				case 'd': d=xx;break;
				case 'e': e=xx;break;
				case 'f': f=xx;break;
				case 'D': D=xx;break;
				case 'E': E=xx;break;
				case 'F': F=xx;break;
			}
		}
		Solve();
	}
	return 0;
}
