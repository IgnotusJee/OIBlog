T1 Candy

数学题。考试时就想到了要算两个数分别的最小质因子，但是该想到式子转化逆元的地方思维就拐弯了，跑去用phthon打表，跑了2h也没跑出来。还是见题太少，导致想偏了。

考虑计算sum(digits)sum(digits)sum(digits)和10n−19\frac{10^n-1}{9}910n−1​的最小质因子。因为前者不会大于5×1065\times10^65×106，所以筛出值域内的所有质数，枚举对后者进行判断。10n−19≡0(modp)⇒10n≡1(mod9×p)\frac{10^n-1}{9} \equiv 0 \pmod p \Rightarrow 10^n \equiv 1 \pmod {9\times p}910n−1​≡0(modp)⇒10n≡1(mod9×p)。快速幂即可。

const int maxn(5e5+10);
int n;
char s[maxn];

inline int QuickPow(int a,int b,const int &mod) {
	int ans(1);
	for(;b;a=a*a%mod,b>>=1) if(b&1) ans=ans*a%mod;
	return ans;
}

int prime[348520],cnt;
int Get1() {
	for(int i(1);i<=cnt;i++) if(QuickPow(10,n,9*prime[i])==1) return prime[i];
	return 0x3f3f3f3f;
}
int Get2(const int &now) {
	if(now==1) return 0x3f3f3f3f;
	for(int i(1);i<=cnt;i++) if(now%prime[i]==0) return prime[i];
	return 0x3f3f3f3f;
}

int main(void) {
#ifndef ONLINE_JUDGE
	freopen("candy.in","r",stdin);
#endif
	static char not_prime[maxn*10];
	not_prime[1]=0xff;
	for(int i(2);i<=5000000;i++) {
		if(~not_prime[i]) prime[++cnt]=i;
		for(int j(1);j<=cnt&&i*prime[j]<=5000000;j++) {
			not_prime[i*prime[j]]=0xff;
			if(i%prime[j]==0) break;
		}
	}
	cin>>(s+1);
	n=strlen(s+1);
	int now(0);
	for(int i(1);i<=n;i++) now+=s[i]-'0';
	cout<<min(Get1(),Get2(now))<<endl;
	return 0;
}

T2 遗失的二叉树

考试时因为T1打表太久，这题就只打了暴搜。没有考虑到其他情况还WA了两个点。后来把暴搜改为记忆化搜索，TLE80pts。

就是一个区间DP。

const int maxn(500+10);
int T,n,a[maxn];

int gcd(const int &a,const int &b) {
	return b ? gcd( b , a%b ) : a;
}

char mp[maxn][maxn],L[maxn][maxn],R[maxn][maxn];
// L[l][r] 表示[l,r]区间以r为根是否有解,R[l][r]表示[l,r]区间以l为根是否有解

int main(void) {
#ifndef ONLINE_JUDGE
	freopen("tree.in","r",stdin);
#endif
	cin>>T;
	while(T--) {
		memset(mp,0x00,sizeof mp);
		memset(L,0x00,sizeof L);
		memset(R,0x00,sizeof R);
		cin>>n;
		for(int i(1);i<=n;i++) cin>>a[i];
		for(int i(1);i<=n;i++)
			for(int j(1);j<=n;j++)
				mp[i][j]|=gcd(a[i],a[j])!=1;
		for(int i(1);i<=n;i++) L[i][i]=R[i][i]=0xff;
		for(int len(2);len<=n;len++)
			for(int l(1);l+len-1<=n;l++) {
				const int r(l+len-1);
				for(int k(l);k<r;k++) L[l][r]|=L[l][k]&R[k][r-1]&mp[r][k];
				for(int k(l+1);k<=r;k++) R[l][r]|=L[l+1][k]&R[k][r]&mp[k][l];
			}
		char flag=0x00;
		for(int i(1);i<=n;i++) flag|=L[1][i]&R[i][n];
		cout<<(flag?"Yes":"No")<<endl;
	}
	return 0;
}

T3 精灵

原题，打出一次便过（假，其实开始没有给复数取模。）

const int maxn(1e5+10),mod(201314);
int n,q;

namespace Tree {
	struct Edge {
		int v,next;
		Edge(const int &v,const int &next) :v(v),next(next) {}
		Edge() {}
	}e[maxn];
	int head[maxn],cnt;
	inline void AddEdge(const int &u,const int &v) {
		e[++cnt]=Edge(v,head[u]);
		head[u]=cnt;
	}

	int size[maxn],son[maxn],depth[maxn],fa[maxn];
	void Dfs1(const int &u=1) {
		size[u]=1;depth[u]=depth[fa[u]]+1;
		for(int i(head[u]);i;i=e[i].next) {
			const int &v(e[i].v);
			fa[v]=u;
			Dfs1(v);
			size[u]+=size[v];
			if(size[v]>size[son[u]]) son[u]=v;
		}
	}
	int dfn[maxn],stamp,top[maxn];
	void Dfs2(const int &u=1,const int &anc=1) {
		dfn[u]=++stamp;top[u]=anc;
		if(son[u]) Dfs2(son[u],anc);
		for(int i(head[u]);i;i=e[i].next) {
			const int &v(e[i].v);
			if(v==son[u]) continue;
			Dfs2(v,v);
		}
	}

	namespace SegmentTree {
	#define lc (a<<1)
	#define rc ((a<<1)|1)
		int val[maxn<<2],delta[maxn<<2];
		inline void pushup(const int &a) {
			val[a]=val[lc]+val[rc];
		}
		inline void pushtag(const int &a,const int &k,const int &l,const int &r) {
			(val[a]+=k*(r-l+1)%mod)%=mod;
			(delta[a]+=k)%=mod;
		}
		inline void pushdown(const int &a,const int &l,const int &r,const int &mid) {
			pushtag(lc,delta[a],l,mid);
			pushtag(rc,delta[a],mid+1,r);
			delta[a]=0;
		}

		void Modify(const int &left,const int &right,const int &k,const int &a=1,const int &l=1,const int &r=n) {
			if(left<=l&&right>=r) return pushtag(a,k,l,r);
			int mid((l+r)>>1);
			if(delta[a]) pushdown(a,l,r,mid);
			if(left<=mid) Modify(left,right,k,lc,l,mid);
			if(right>mid) Modify(left,right,k,rc,mid+1,r);
			pushup(a);
		}
		int Query(const int &left,const int &right,const int &a=1,const int &l=1,const int &r=n) {
			if(left<=l&&right>=r) return val[a]%mod;
			int mid((l+r)>>1);
			if(delta[a]) pushdown(a,l,r,mid);
			if(right<=mid) return Query(left,right,lc,l,mid);
			if(left>mid) return Query(left,right,rc,mid+1,r);
			return (Query(left,right,lc,l,mid)+Query(left,right,rc,mid+1,r))%mod;
		}
	#undef lc
	#undef rc
	}

	inline void Modify(int u) {
		for(;u;u=fa[top[u]]) SegmentTree::Modify(dfn[top[u]],dfn[u],1);
	}
	inline int Query(int u) {
		int ans(0);
		for(;u;u=fa[top[u]]) (ans+=SegmentTree::Query(dfn[top[u]],dfn[u]))%=mod;
		return ans;
	}
}

int ans[maxn];
struct QueryNode {
	int z,id;
	QueryNode(const int &z,const int &id) :z(z),id(id) {}
};
std::vector<QueryNode> v[maxn];
int main(void) {
#ifndef ONLINE_JUDGE
	freopen("elf.in","r",stdin);
	freopen("elf.out","w",stdout);
#endif
	cin>>n>>q;
	for(int i(2);i<=n;i++) {
		int x;cin>>x;
		Tree::AddEdge(x,i);
	}
	Tree::Dfs1();Tree::Dfs2();
	for(int i(1);i<=q;i++) {
		int l,r,z;
		cin>>l>>r>>z;
		v[l-1].emplace_back(z,-i);
		v[r].emplace_back(z,i);
	}
	for(int i(1);i<=n;i++) {
		Tree::Modify(i);
		for(const auto &p: v[i])
			(ans[abs(p.id)]+=((p.id>0?1:-1)*Tree::Query(p.z)+mod)%mod)%=mod;
	}
	for(int i(1);i<=q;i++) cout<<ans[i]<<endl;
	return 0;
}
