T1 开方

垃圾题，直接二分出边界（其实就是225=2322^{2^5}=2^{32}225=232），然后判断。但题面很鬼，没有定义y=0y=0y=0的情况。希望出题人没事

const int maxn(100+10);
using ll = long long;

inline int Times(ll x) {
	int cnt(0);
	for(;x&&x!=1LL;x=sqrt(x)) cnt++;
	return cnt;
}

inline int Solve(char *s,const int &len) {
	if(len>10) return 6;
	ll x(0);
	for(int i(1);i<=len;i++) x=(x<<1LL)+(x<<3LL)+ll(s[i]^48);
	if(x>4294967295LL||!x) return 6;
	if(x==1LL) return 0;
	return Times(x);
}

int main(void) {
	static char s[maxn];
	while(cin>>(s+1)) {
		int ans(Solve(s,strlen(s+1)));
		if(ans==6) cout<<"TAT"<<endl;
		else cout<<ans<<endl;
	}
	return 0;
}

T2 染色

考试时想了先给边排序，然后枚举选多边形的解法，开始以为复杂度是假的，就迟迟没打。后来想到实际上只会有n−1n-1n−1条边有用，所以一般跑不到n2n^2n2上界，于是就直接上了。

交上去T了4个点说好的跑不到上界呢， 于是只能优化。先裸跑一次最小生成树，发现其实只有裸跑时的边有用，所以只遍历原最小生成树的n−1n-1n−1条边。优化了一维时间，这样就过了。

但为什么题解prim复杂度n3n^3n3还能过啊。

const int maxn(2000+10),maxm(13);
int n,m;
using ll = long long;
const ll Inf(0x3f3f3f3f3f3f3f3f);

class UnionAndFindSet {
private:
	int fa[maxn],rank[maxn];
	int Find(const int &x) {
		return x == fa[x] ? x : fa[x] = Find(fa[x]);
	}
public:
	inline void Init() {
		for(int i(1);i<=n;i++) fa[i]=i,rank[i]=1;
	}
	inline bool Merge(const int &x,const int &y) {
		int u(Find(x)),v(Find(y));
		if(u==v) return false;
		if(rank[u]<rank[v]) swap(u,v);
		fa[v]=u;rank[u]=max(rank[u],rank[v]+1);
		return true;
	}
}U;

ll x[maxn],y[maxn];
inline ll Cost(const int &i,const int &j) {
	return (x[i]-x[j])*(x[i]-x[j])+(y[i]-y[j])*(y[i]-y[j]);
}

struct Edge {
	int u,v;ll w;
	Edge(const int &u,const int &v,const ll &w) :u(u),v(v),w(w) {}
	Edge() {}
}e[maxn*maxn],g[maxn];
int Cnt_Edge1,Cnt_Edge2;
ll c[maxm];
std::vector<int> v[maxm];

void Kruscal0() {
	U.Init();
	for(int i(1);i<=Cnt_Edge1&&Cnt_Edge2<n;i++)
		if(U.Merge(e[i].u,e[i].v))
			g[++Cnt_Edge2]=e[i];
}

inline ll Kruscal(const int &s) {
	U.Init();
	int cnt(0);ll ans(0);
	for(int i(1);i<=m;i++)
		if(s&(1<<(i-1))) {
			for(const int &p: v[i])
				cnt+=U.Merge(p,v[i][0]);
			ans+=c[i];
		}
	for(int i(1);i<=Cnt_Edge2&&cnt<n;i++) {
		if(!U.Merge(g[i].u,g[i].v)) continue;
		ans+=g[i].w;
		cnt++;
	}
	return ans;
}

int main(void) {
	cin>>n>>m;
	for(int i(1);i<=n;i++) cin>>x[i]>>y[i];
	for(int i(1);i<=m;i++) {
		int k;cin>>k>>c[i];
		for(int j(1);j<=k;j++) {
			int x;cin>>x;
			v[i].push_back(x);
		}
	}
	for(int i(1);i<=n;i++)
		for(int j(i+1);j<=n;j++)
			e[++Cnt_Edge1]=Edge(i,j,Cost(i,j));
	std::sort(e+1,e+1+Cnt_Edge1,[](const Edge &p,const Edge &q) {
		return p.w<q.w;
	});
	Kruscal0();
	ll ans(Inf);
	for(int i(0);i<(1<<m);i++)
		ans=min(ans,Kruscal(i));
	cout<<ans<<endl;
	return 0;
}

T3 心情

考试时先想到可能是区间DP，但算了三位数组的空间卡的很满就转向想贪心了，结果只得了40pts。其实就是区间DP。

const int maxn(300+10),nef(0xc1c1c1c1);
int n,a[maxn];
int col[maxn],len[maxn],last[maxn],m;// last[i] 表示与i颜色段相同颜色的前一个颜色段
int f[maxn][maxn][maxn];
// f[i][j][k] 表示从i段到j段，后面接上长度为k的与j颜色相同段的最大花费

int Dfs(const int &i,const int &j,const int &k) {
	if(i>j) return 0;
	if(~f[i][j][k]) return f[i][j][k];
	f[i][j][k]=nef;
	f[i][j][k]=max(f[i][j][k],Dfs(i,j-1,0)+(len[j]+k)*(len[j]+k));
	for(int p(last[j]);p>=i;p=last[p])
		f[i][j][k]=max(f[i][j][k],Dfs(i,p,len[j]+k)+Dfs(p+1,j-1,0));
	return f[i][j][k];
}

int main(void) {
	cin>>n;
	for(int i(1);i<=n;i++) {
		cin>>a[i];
		if(a[i]==col[m]) len[m]++;
		else m++,col[m]=a[i],len[m]=1;
	}
	static int tax[maxn];
	for(int i(1);i<=m;i++) {
		last[i]=tax[col[i]];
		tax[col[i]]=i;
	}
	memset(f,0xff,sizeof f);
	cout<<Dfs(1,n,0)<<endl;
	return 0;
}
